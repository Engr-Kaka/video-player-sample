 17 /**
 18  * Provide a namespace for this class
 19  */
 20 goog.provide('gva.util.DateUtil');
 21 
 22 /**
 23  * A set of static utility methods for formatting Date objects in a manner similar to PHP
 24  * Based on: http://jacwright.com/projects/javascript/date_format/
 25  */
 26 gva.util.DateUtil = (function() {
 27 
 28     var _shortMonths = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
 29     var _longMonths = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December'];
 30     var _shortDays = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];
 31     var _longDays = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'];
 32 
 33     var _methods = {
 34          // Day
 35         'd': function(_d) { return (_d.getDate() < 10 ? '0' : '') + _d.getDate(); },
 36         'D': function(_d) { return _shortDays[_d.getDay()]; },
 37         'j': function(_d) { return _d.getDate(); },
 38         'l': function(_d) { return _longDays[_d.getDay()]; },
 39         'N': function(_d) { return _d.getDay() + 1; },
 40         'S': function(_d) { return (_d.getDate() % 10 == 1 && _d.getDate() != 11 ? 'st' : (_d.getDate() % 10 == 2 && _d.getDate() != 12 ? 'nd' : (_d.getDate() % 10 == 3 && _d.getDate() != 13 ? 'rd' : 'th'))); },
 41         'w': function(_d) { return _d.getDay(); },
 42         'z': function(_d) { var d = new Date(_d.getFullYear(), 0, 1); return Math.ceil((this - d) / 86400000); }, // Fixed now
 43         // Week
 44         'W': function(_d) { var d = new Date(_d.getFullYear(), 0, 1); return Math.ceil((((this - d) / 86400000) + d.getDay() + 1) / 7); }, // Fixed now
 45         // Month
 46         'F': function(_d) { return _longMonths[_d.getMonth()]; },
 47         'm': function(_d) { return (_d.getMonth() < 9 ? '0' : '') + (_d.getMonth() + 1); },
 48         'M': function(_d) { return _shortMonths[_d.getMonth()]; },
 49         'n': function(_d) { return _d.getMonth() + 1; },
 50         't': function(_d) { var d = new Date(); return new Date(d.getFullYear(), d.getMonth(), 0).getDate();}, // Fixed now, gets #days of date
 51         // Year
 52         'L': function(_d) { var year = _d.getFullYear(); return (year % 400 == 0 || (year % 100 != 0 && year % 4 == 0)); },             // Fixed now
 53         'o': function(_d) { var d = new Date(_d.valueOf()); d.setDate(d.getDate() - ((_d.getDay() + 6) % 7) + 3); return d.getFullYear();}, //Fixed now
 54         'Y': function(_d) { return _d.getFullYear(); },
 55         'y': function(_d) { return ('' + _d.getFullYear()).substr(2); },
 56         // Time
 57         'a': function(_d) { return _d.getHours() < 12 ? 'am' : 'pm'; },
 58         'A': function(_d) { return _d.getHours() < 12 ? 'AM' : 'PM'; },
 59         'B': function(_d) { return Math.floor((((_d.getUTCHours() + 1) % 24) + _d.getUTCMinutes() / 60 + _d.getUTCSeconds() / 3600) * 1000 / 24); }, // Fixed now
 60         'g': function(_d) { return _d.getHours() % 12 || 12; },
 61         'G': function(_d) { return _d.getHours(); },
 62         'h': function(_d) { return ((_d.getHours() % 12 || 12) < 10 ? '0' : '') + (_d.getHours() % 12 || 12); },
 63         'H': function(_d) { return (_d.getHours() < 10 ? '0' : '') + _d.getHours(); },
 64         'i': function(_d) { return (_d.getMinutes() < 10 ? '0' : '') + _d.getMinutes(); },
 65         's': function(_d) { return (_d.getSeconds() < 10 ? '0' : '') + _d.getSeconds(); },
 66         'u': function(_d) { var m = _d.getMilliseconds(); return (m < 10 ? '00' : (m < 100 ? '0' : '')) + m; },
 67         // Timezone
 68         'e': function(_d) { return 'Not Yet Supported'; },
 69         'I': function(_d) { return 'Not Yet Supported'; },
 70         'O': function(_d) { return (-_d.getTimezoneOffset() < 0 ? '-' : '+') + (Math.abs(_d.getTimezoneOffset() / 60) < 10 ? '0' : '') + (Math.abs(_d.getTimezoneOffset() / 60)) + '00'; },
 71         'P': function(_d) { return (-_d.getTimezoneOffset() < 0 ? '-' : '+') + (Math.abs(_d.getTimezoneOffset() / 60) < 10 ? '0' : '') + (Math.abs(_d.getTimezoneOffset() / 60)) + ':00'; }, // Fixed now
 72         'T': function(_d) { var m = _d.getMonth(); _d.setMonth(0); var result = _d.toTimeString().replace(/^.+ \(?([^\)]+)\)?$/, '$1'); _d.setMonth(m); return result;},
 73         'Z': function(_d) { return -_d.getTimezoneOffset() * 60; },
 74         // Full Date/Time
 75         'c': function(_d) { return _d.formatDate('Y-m-d\\TH:i:sP'); },
 76         'r': function(_d) { return _d.toString(); },
 77         'U': function(_d) { return _d.getTime() / 1000; }
 78     };
 79 
 80     return {
 81 
 82         /**
 83          * Returns a string representation of a Date object based on the provided template
 84          * @param {Date} date The Date to format.
 85          * @param {string} format The format to return the Date in.
 86          * @return {string} The string representation of the Date.
 87          */
 88         formatDate: function(date, format) {
 89             var result = '', str;
 90             for (var i = 0; i < format.length; i++) {
 91                 str = format.charAt(i);
 92                 if (i - 1 >= 0 && format.charAt(i - 1) == '\\') {
 93                     result += str;
 94                 }
 95                 else if (_methods[str]) {
 96                     result += _methods[str](date);
 97                 } else if (str != '\\') {
 98                     result += str;
 99                 }
100             }
101             return result;
102         }
103     };
104 
105 })();
