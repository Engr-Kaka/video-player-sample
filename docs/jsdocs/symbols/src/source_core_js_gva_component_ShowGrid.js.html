 47 /**
 48  * Custom DragDropGroup implementation to override drag positioning.
 49  * @constructor
 50  * @extends {goog.fx.DragDropGroup}
 51  */
 52 gva.ShowDragDropGroup = function() {
 53   goog.fx.DragDropGroup.call(this);
 54 };
 55 goog.inherits(gva.ShowDragDropGroup, goog.fx.DragDropGroup);

 58 /**
 59 * (Overrides goog.fx.DragDropGroup.getDragElementPosition)
 60 * Returns the position for the drag element.
 61 *
 62 * @param {Element} sourceEl Drag source element.
 63 * @param {Element} el The dragged element created by createDragElement().
 64 * @param {goog.events.BrowserEvent} event Mouse down event for start of drag.
 65 * @return {goog.math.Coordinate} The position for the drag element.
 66 */
 67 gva.ShowDragDropGroup.prototype.getDragElementPosition = function(sourceEl, el, event) {
 68   return new goog.math.Coordinate(event.clientX - 100, event.clientY - 150);
 69 };

 71 /**
 72  * @constructor
 73  * @extends {gva.component.BaseComponent}
 74  */
 75 gva.component.ShowGrid = function() {
 76 
 77     /**
 78      * The flexible grid for arranging items.
 79      * @type {gva.component.FlexibleGrid}
 80      */
 81     this._grid = new gva.component.FlexibleGrid();
 82 
 83     // The model instances.
 84     this._userModel = null;
 85     this._dataModel = null;
 86 
 87     // The subscriptions component instance.
 88     this._subscriptions = null;
 89 
 90     /**
 91      * The object containing all draggable shows.
 92      * @type {goog.fx.DragDropGroup}
 93      */
 94     this._draggableShows = null;
 95 
 96     /**
 97      * The drop area for draggable shows.
 98      * @type {goog.fx.DragDrop}
 99      */
100     this._dropArea = null;
101 
102     // Boolean for whether or not a show is being dragged.
103     this.dragging = false;
104 
105     // Timeouts for changing back to default text and closing the 'show added' text.
106     this.confirmationTextTimer = null;
107     this.confirmationCloseTimer = null;
108 
109     // Timeout for removing the up state of the subscriptions panel
110     this.closeSubscriptionsTimer = null;
111 
112     // Duration (ms) of show added confirmation.
113     this.CONFIRMATION_DURATION = 2000;
114 
115     // Default text for the droppable area.
116     this.DROP_INSTRUCTIONS = 'Drop Shows here to subscribe';
117 
118     // Whether or not a successful drop has occurred recently.
119     // Should be set to false by dragend event immediately after drop event.
120     this.recentlyDropped = false;
121 
122     // Call super constructor
123     gva.component.BaseComponent.call(this);
124 };

148 /**
149  * Creates a DOM element for an item and appends it to the grid
150  * @param {gva.vo.Show} item The data for the item to add.
151  */
152 gva.component.ShowGrid.prototype.addItem = function(item) {
153 
154     // Format the date
155     var date = new Date(item.latestvideodate);
156     var dateStr = gva.util.DateUtil.formatDate(date, 'M j, Y');
157 
158     var subscribeClass = item.subscribed ? 'subscribed' : '';
159     var subscribeCopy = item.subscribed ? 'Remove' : 'Subscribe';
160 
161     var episodesLink = '/shows/' + item.slug + '/';
162 
163     // Create markup for the grid item
164     var itemMarkup = '' +
165     '<article class="show" data-showid="' + item.id + '">' +
166         '<figure>' +
167             '<img src="' + item.image_url + '"/>' +
168         '</figure>' +
169         '<header>' +
170             '<h3>' + item.title + '</h3>' +
171         '</header>' +
172         '<p>' + item.description + '</p>' +
173         '<footer>' +
174             '<span class="icon"></span><p>' + item.videocount + ' Videos, Updated: ' + dateStr + '</p>' +
175         '</footer>' +
176         '<nav>' +
177             '<a class="subscribe ' + subscribeClass + '" href="#" data-showid="' + item.id + '">' +
178                 '<strong><span class="icon"></span><em>' + subscribeCopy + '</em></strong>' +
179             '</a>' +
180             '<a class="episodes" title="' + item.title + '" href="' + episodesLink + '"><strong><span class="icon"></span><em>Episodes</em></strong></a>' +
181         '</nav>' +
182     '</article>';
183 
184     var $item = null;
185 
186     if (goog.userAgent.IE && goog.userAgent.VERSION < 9) {
187       // Create show element for older browsers.
188       $item = innerShiv(itemMarkup, false)[0];
189     }
190     else {
191       // Create a DOM element from the markup fragment
192       $item = goog.dom.htmlToDocumentFragment(itemMarkup);
193     }
194 
195     var $subscribeBtn = goog.dom.query('a.subscribe', $item)[0];
196     var $episodesBtn = goog.dom.query('a.episodes', $item)[0];
197 
198     // Bind event handlers to the object
199     goog.events.listen($subscribeBtn, goog.events.EventType.CLICK, goog.bind(this.onSubscribeClicked, this));
200     goog.events.listen($episodesBtn, goog.events.EventType.CLICK, goog.bind(this.onEpisodesClicked, this));
201 
202     // Add the item to the grid
203     this._grid.addItem($item);
204 };

206 /**
207  * Refreshes the grid and updates the items based on the latest data
208  * @param {Array.<Object>} showData The updated show data.
209  */
210 gva.component.ShowGrid.prototype.refresh = function(showData) {
211 
212     var $gridItems = this._grid.getItems();
213     var $gridItem;
214     var $subscribeBtn;
215     var $linkText;
216     var showObj;
217     var showID;
218 
219     for (var i = 0, n = $gridItems.length; i < n; i++) {
220 
221         $gridItem = $gridItems[i];
222 
223         // Get the show ID from the elements data attribute
224         showID = goog.dom.dataset.get($gridItem, 'showid');
225 
226         // Find the corresponding show data object
227         showObj = goog.array.find(showData, function(show) {
228             return show.id === showID;
229         });
230 
231         // Update the subscribe button based on the new data
232         $subscribeBtn = goog.dom.query('a.subscribe', $gridItem)[0];
233         $linkText = goog.dom.query('em', $subscribeBtn)[0];
234 
235         // Update the text content
236         if (showObj && showObj.subscribed) {
237             goog.dom.classes.add($subscribeBtn, 'subscribed');
238             goog.dom.setTextContent($linkText, 'Remove');
239         } else {
240             goog.dom.classes.remove($subscribeBtn, 'subscribed');
241             goog.dom.setTextContent($linkText, 'Subscribe');
242         }
243     }
244 };

246 /**
247  * Event handler for when an item's subscribe button is clicked
248  * @param {goog.events.Event} event The event which triggered the handler.
249  */
250 gva.component.ShowGrid.prototype.onSubscribeClicked = function(event) {
251 
252     event.preventDefault();
253 
254     var $item = event.currentTarget;
255     var showID = goog.dom.dataset.get($item, 'showid');
256     var subscribed = goog.dom.classes.has($item, 'subscribed');
257 
258     // Add or remove the subscription based on the current state
259     if (subscribed) {
260         this._userModel.removeUserSubscription(showID);
261     } else {
262         this._userModel.addUserSubscription(showID);
263     }
264 };

279 };

282 /**
283  * Initializes the draggable shows and the subscription drop target.
284  */
285 gva.component.ShowGrid.prototype.initDragDrop = function() {
286 
287   // Get subscription component instance.
288   this._subscriptions = gva.component.ComponentInitializer.getComponentByQuery('#subscriptions');
289 
290   // Get show articles.
291   var $gridItems = this._grid.getItems();
292 
293   // Initialize DragDropGroup
294   this._draggableShows = new gva.ShowDragDropGroup();
295 
296   // Add draggable items.
297   for (var i = 0, n = $gridItems.length; i < n; i++) {
298 
299     var $gridItem = $gridItems[i];
300 
301     var $draggableItem = goog.dom.getElementsByTagNameAndClass('figure', null, $gridItem)[0];
302 
303     // Get data for the draggable to carry.
304     var showID = goog.dom.dataset.get($gridItem, 'showid');
305     var show = this._dataModel.getShowByID(showID);
306     var data = {};
307     data['title'] = show.title;
308     data['showID'] = showID;
309 
310     this._draggableShows.addItem($draggableItem, data);
311 
312     // Set hover states for revealing droppable area/instructions.
313     goog.events.listen($gridItem, goog.events.EventType.MOUSEOVER, this.onDraggableMouseOver, false, this);
314     goog.events.listen($gridItem, goog.events.EventType.MOUSEOUT, this.onDraggableMouseOut, false, this);
315   }
316 
317   // Init droppable area and set as target for draggables.
318   var $dropAreaEl = this._subscriptions.$actionIndicator;
319   this._dropArea = new goog.fx.DragDrop($dropAreaEl);
320 
321   this._draggableShows.addTarget(this._dropArea);
322 
323   // Add DragDrop classes
324   this._draggableShows.setSourceClass('source');
325   this._draggableShows.setDragClass('dragging');
326 
327   // Init to add listeners and start the drag/drop action.
328   this._draggableShows.init();
329 
330   // Set custom dragdrop listeners.
331   goog.events.listen(this._draggableShows, goog.fx.AbstractDragDrop.EventType.DRAGSTART, goog.bind(this.onDragStart, this), false, this);
332   goog.events.listen(this._draggableShows, goog.fx.AbstractDragDrop.EventType.DRAGEND, goog.bind(this.onDragEnd, this), false, this);
333   goog.events.listen(this._dropArea, goog.fx.AbstractDragDrop.EventType.DROP, goog.bind(this.onDrop, this), false, this);
334 };

336 /**
337  * Event handler for when a show starts being dragged.
338  * @param {goog.events.Event} event The event which triggered the handler.
339  */
340 gva.component.ShowGrid.prototype.onDragStart = function(event) {
341   this.toggleDropArea(true);
342   this.dragging = true;

346 /**
347  * Event handler for when a show finishes being dragged.
348  * @param {goog.events.Event} event The event which triggered the handler.
349  */
350 gva.component.ShowGrid.prototype.onDragEnd = function(event) {
351   this.dragging = false;
352 
353   if (!this.recentlyDropped) {
354     this.toggleDropArea(false);
355   }
356 
357   this.recentlyDropped = false;
358 };

361 /**
362  * Event handler for when a show is dropped onto the target.
363  * @param {goog.events.Event} event The event which triggered the handler.
364  */
365 gva.component.ShowGrid.prototype.onDrop = function(event) {
366 
367   var showData = event.dragSourceItem.data;
368 
369   // Determine if already subscribed.
370   var subscribed = false;
371   var subscriptions = this._userModel.getUserSubscriptions();
372 
373   if (goog.array.indexOf(subscriptions, showData['showID']) != -1) {
374     subscribed = true;
375   }
376 
377   // Add or remove the subscription based on the current state
378   if (!subscribed) {
379     this.recentlyDropped = true;
380     this._userModel.addUserSubscription(showData['showID']);
381     this.showDropConfirmation(showData['title']);
382   }
383 };

386 /**
387  * Slides the subscription button up and down to reveal the droppable area.
388  * @param {Boolean} display Set to true to reveal the droppable area and false to close it.
389  */
390 gva.component.ShowGrid.prototype.toggleDropArea = function(display) {
391 
392   clearTimeout(this.closeSubscriptionsTimer);
393 
394   if (display) {
395     goog.dom.classes.add(this._subscriptions.$context, 'up');
396   }
397   else if (!this.dragging) {
398     var self = this;
399     this.closeSubscriptionsTimer = setTimeout(function() {
400       goog.dom.classes.remove(self._subscriptions.$context, 'up');
401     },500);
402   }
403 };

405 /**
406  * Updates the action indicator text/icon to display a confirmation t=emporarily.
407  * @param {String} showName The name of the show that has been added.
408  */
409 gva.component.ShowGrid.prototype.showDropConfirmation = function(showName) {
410 
411   this.resetTimers();
412 
413   goog.dom.classes.add(this._subscriptions.$actionIndicator, 'confirm-addition');
414   this._subscriptions.$actionStatus.innerHTML = showName + ' added';
415 
416   this.confirmationTextTimer = setTimeout(goog.bind(this.hideDropConfirmation, this), this.CONFIRMATION_DURATION);
417   this.confirmationCloseTimer = setTimeout(goog.bind(function() {
418     this.toggleDropArea(false);
419     this.confirmationCloseTimer = null;
420   }, this), this.CONFIRMATION_DURATION - 250);
421 };
422 
423 
424 /**
425  * Updates the action indicator text/icon to default display.
426  */
427 gva.component.ShowGrid.prototype.hideDropConfirmation = function() {
428   goog.dom.classes.remove(this._subscriptions.$actionIndicator, 'confirm-addition');
429   this._subscriptions.$actionStatus.innerHTML = this.DROP_INSTRUCTIONS;
430 };
