 38 /**
 39  * @constructor
 40  * @extends {gva.controller.AbstractController}
 41  */
 42 gva.controller.ShowController = function() {
 43 
 44     this._showGrid = null;
 45     this._showDetails = null;
 46     this._episodeGrid = null;
 47     /**
 48      * @type {Array.<gva.vo.Show>}
 49      */
 50     this._showData = null;
 51     this._currentShow = null;
 52     /**
 53      * @type {gva.vo.EpisodesData}
 54      */
 55     this._episodeData = null;
 56     this._awaitingData = false;
 57     this._urlSegments = null;
 58 
 59     this._transition = null;
 60     this._subsection = true;
 61 
 62     this._$showTitle = null;
 63     this._$backBtn = null;
 64 
 65     this._callbacks = null;
 66 
 67     // Call super constructor
 68     gva.controller.AbstractController.call(this);
 69 };

 76 /**
 77  * @inheritDoc
 78  */
 79 gva.controller.ShowController.prototype.initialize = function($context) {
 80 
 81     // Call super initialize method
 82     this.constructor.superClass_.initialize.call(this, $context);
 83 
 84     // Find the heading which we'll populate when browsing show information
 85     this._$showTitle = goog.dom.query('.details header h2', this.$context)[0];
 86     this._$backBtn = goog.dom.query('.details a.back', this.$context)[0];
 87 
 88     // Retrieve component references from the component initializer
 89     this._showGrid = gva.component.ComponentInitializer.getComponentByQuery('.show-grid');
 90     this._episodeGrid = gva.component.ComponentInitializer.getComponentByQuery('.episode-grid');
 91     this._showDetails = gva.component.ComponentInitializer.getComponentByQuery('.show-details');
 92 
 93     // Bind events
 94     this.addEventListeners();
 95 };

 98 /**
 99  * Adds event listeners for show components.
100  */
101 gva.controller.ShowController.prototype.addEventListeners = function() {
102 
103     // Create scoped callbacks
104     this._callbacks = {
105             onAllDataLoaded: goog.bind(this.onAllDataLoaded, this),
106             onShowDataLoaded: goog.bind(this.onShowDataLoaded, this),
107             onEpisodeDataLoaded: goog.bind(this.onEpisodeDataLoaded, this),
108             onShowDetailsResize: goog.bind(this.onShowDetailsResize, this),
109             onSubscriptionChanged: goog.bind(this.onSubscriptionChanged, this),
110             onEpisodeSelected: goog.bind(this.onEpisodeSelected, this),
111             onEpisodeWatched: goog.bind(this.onEpisodeWatched, this),
112             onBackClicked: goog.bind(this.onBackClicked, this)
113     };
114 
115     // Retrieve the model singletons
116     var dataModel = gva.model.DataModel.getInstance();
117     var userModel = gva.model.UserModel.getInstance();
118 
119     // Listen for back button clicks
120     goog.events.listen(this._$backBtn, goog.events.EventType.CLICK, this._callbacks.onBackClicked);
121 
122     // Listen for height change events on the details component
123     this._showDetails.addEventListener(gva.component.ShowDetails.EventType.RESIZE, this._callbacks.onShowDetailsResize, false, this);
124 
125 
126     // Retrieve show and episode data from the model
127     this._callbacks.onAllDataLoaded();
128 
129     // Listen for when a show is subscribed to or unsubscribed from
130     goog.events.listen(userModel, gva.model.UserModel.EventType.SUBSCRIPTION_ADDED, this._callbacks.onSubscriptionChanged);
131     goog.events.listen(userModel, gva.model.UserModel.EventType.SUBSCRIPTION_REMOVED, this._callbacks.onSubscriptionChanged);
132     goog.events.listen(userModel, gva.model.UserModel.EventType.EPISODE_WATCHED, this._callbacks.onEpisodeWatched);
133 };

151 /**
152  * Tells the controller to handle deep linking beyond the top level section depth.
153  * @param {Array.<string>} segments A list of URL fragments beyond the root level.
154  * @param {boolean=} instant If true, no transition will be applied.
155  */
156 gva.controller.ShowController.prototype.navigate = function(segments, instant) {
157 
158     this._urlSegments = segments;
159 
160     // Create a recyclable transition animation
161     if (!this._transition) {
162 
163         this._transition = new goog.fx.Animation([0], [1], 450, gva.easing.quintic.easeOut);
164 
165         var events = [];
166 
167         // Subscribe to the progress and end animation events
168         events.push(goog.fx.Animation.EventType.ANIMATE);
169         events.push(goog.fx.Animation.EventType.END);
170 
171         var self = this;
172 
173         goog.events.listen(this._transition, events, function() {
174 
175             // Calculate where the container should be
176             var percent = self._transition.coords[0] * 100;
177             if (!self._subsection) {
178                 percent = 100 - percent;
179             }
180 
181             // Update the container position
182             goog.style.setStyle(self.$context, 'left', '-' + percent + '%');
183         });
184     }
185 
186     // Set the transition duration
187     this._transition.duration = instant ? 1 : 450;
188 
189     // Set the direction of the animation
190     this._subsection = segments.length > 0;
191 
192     if (this._subsection) {
193 
194         var model = gva.model.DataModel.getInstance();
195         var show = model.getShowBySlug(segments[0]);
196 
197         if (show) {
198 
199             // Store the current show
200             this._currentShow = show;
201 
202             // Update the section title and show information
203             goog.dom.setTextContent(this._$showTitle, show.title);
204             this._showDetails.setShow(show);
205 
206             // Filter episodes by show and pass to the episode grid
207             if (this._episodeData) {
208                 this.refreshEpisodeData();
209             } else {
210                 this._awaitingData = true;
211             }
212 
213             // Enable the back button
214             var $backBtn = this._$backBtn;
215             setTimeout(function() {
216               goog.dom.classes.add($backBtn, 'enabled');
217             }, 500);
218 
219             // Retrieve the episode for this slug, if present
220             if (segments[1]) {
221                 var episode = model.getEpisodeBySlug(segments[1]);
222             }
223 
224         } else {
225             //console.log('invalid show URL');
226         }
227 
228     } else {
229         // Disable the back button
230         goog.dom.classes.remove(this._$backBtn, 'enabled');
231     }
232 
233     // Play the animation
234     this._transition.play(true);
235 };

237 /**
238  * Refreshes the episode grid with the current episode data
239  */
240 gva.controller.ShowController.prototype.refreshEpisodeData = function() {
241 
242     if (this._currentShow) {
243 
244         var showEpisodes = [];
245         var allEpisodes = this._episodeData.episodes;
246         for (var i = 0, n = allEpisodes.length; i < n; i++) {
247             if (allEpisodes[i].show_id === this._currentShow.id) {
248                 showEpisodes.push(allEpisodes[i]);
249             }
250         }
251 
252         this._episodeGrid.setEpisodes(showEpisodes);
253     }
254 };

256 /**
257  * Event handler for the resize event dispatched by the show details component
258  * @param {goog.events.Event} event The event which triggered this handler.
259  */
260 gva.controller.ShowController.prototype.onShowDetailsResize = function(event) {
261     // Push the grid down by the current height of the details container
262     var offset = this._showDetails.currentHeight;
263     if (offset > 0) {
264         offset -= 25;
265     }
266     goog.style.setStyle(this._episodeGrid.$context, 'margin-top', offset + 'px');
267 };

270 /**
271  * Event handler for when all episode/show data is ready.
272  * @param {goog.events.Event} event The event which triggered this handler.
273  */
274 gva.controller.ShowController.prototype.onAllDataLoaded = function(event) {
275 
276     // Retrieve data from the model
277     var model = gva.model.DataModel.getInstance();
278     this._showData = model.showsData.shows;
279     this._episodeData = model.episodesData;
280 
281     for (var i = 0, n = this._showData.length; i < n; i++) {
282         this._showGrid.addItem(this._showData[i]);
283     }
284 
285     // Initialize drag/drop elements now that all shows have been loaded.
286     this._showGrid.initDragDrop();
287 
288     // Initialise episodes
289     if (this._awaitingData) {
290         this.refreshEpisodeData();
291     }
292 };

294 /**
295  * Event handler for the show data loaded event, dispatched from the data model
296  * @param {goog.events.Event} event An event containing show data from the data model.
297  */
298 gva.controller.ShowController.prototype.onShowDataLoaded = function(event) {
299     // Add every show to the show grid
300     this._showData = event.data.shows;
301     for (var i = 0, n = this._showData.length; i < n; i++) {
302         this._showGrid.addItem(this._showData[i]);
303     }
304 
305     // Initialize drag/drop elements now that all shows have been loaded.
306     this._showGrid.initDragDrop();
307 };
308 
309 /**
310  * Event handler for the episode data loaded event, dispatched from the data model
311  * @param {goog.events.Event} event An event containing episode data from the data model.
312  */
313 gva.controller.ShowController.prototype.onEpisodeDataLoaded = function(event) {
314     this._episodeData = event.data;
315     if (this._awaitingData) {
316         this.refreshEpisodeData();
317     }
318 };

320 /**
321  * Event handler for a show subscription changing in the user model
322  * @param {goog.events.Event} event An event containing the show which has changed.
323  */
324 gva.controller.ShowController.prototype.onSubscriptionChanged = function(event) {
325 
326     // Refresh the show grid and it's data
327     this._showGrid.refresh(this._showData);
328 
329     // Refresh the show details view
330     this._showDetails.refresh();
331 };
332 
333 /**
334  * Event handler for an episode being selected from the episode grid
335  * @param {goog.events.Event} event An event containing the episode ID of the selected episode.
336  */
337 gva.controller.ShowController.prototype.onEpisodeSelected = function(event) {
338     var model = gva.model.DataModel.getInstance();
339     var episode = model.getEpisodeByID(event.id);
340     gva.controller.ApplicationController.navigate('/home/' + episode.slug + '/');
341 };
342 
343 /**
344  * Event handler for an episode being marked as watched in the user model
345  * @param {goog.events.Event} event An event containing the episode data of the watched episode.
346  */
347 gva.controller.ShowController.prototype.onEpisodeWatched = function(event) {
348     // Refresh the episode grid and it's data
349     this._episodeGrid.refresh(this._episodeData.episodes);
350 };
351 
352 /**
353  * Event handler for the back to show button click event
354  * @param {goog.events.BrowserEvent} event The event which triggered this handler.
355  */
356 gva.controller.ShowController.prototype.onBackClicked = function(event) {
357     event.preventDefault();
358     // Go back to shows section
359     gva.controller.ApplicationController.navigate('/shows/');
360 };
