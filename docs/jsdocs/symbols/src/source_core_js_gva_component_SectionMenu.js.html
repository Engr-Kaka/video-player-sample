 17 /**
 18  * Provide a namespace for this class
 19  */
 20 goog.provide('gva.component.SectionMenu');
 21 
 22 /**
 23  * Import dependencies
 24  */
 25 goog.require('goog.array');
 26 goog.require('goog.dom');
 27 goog.require('goog.events');
 28 goog.require('goog.events');
 29 goog.require('goog.events.EventType');
 30 goog.require('gva.component.BaseComponent');
 31 
 32 /**
 33  * @constructor
 34  * @extends {gva.component.BaseComponent}
 35  */
 36 gva.component.SectionMenu = function() {
 37 
 38     /**
 39      * An array of all section links in the menu
 40      * @type {Array.<Element>}
 41      */
 42     this.$menuItems = null;
 43 
 44     // Call super constructor.
 45     gva.component.BaseComponent.call(this);
 46 };
 47 
 48 /**
 49  * Extend gva.component.BaseComponent.
 50  */
 51 goog.inherits(gva.component.SectionMenu, gva.component.BaseComponent);
 52 
 53 /**
 54  * @inheritDoc
 55  */
 56 gva.component.SectionMenu.prototype.initialize = function($context) {
 57 
 58     // Call super method.
 59     this.constructor.superClass_.initialize.call(this, $context);
 60 
 61     // Get all menu items inside the context.
 62     this.$menuItems = goog.dom.getElementsByTagNameAndClass('a', null, $context);
 63 
 64     // Proxy for click handler.
 65     var clickHandler = goog.bind(this.onMenuItemClicked, this);
 66 
 67     // Bind click handlers to navigation items.
 68     goog.array.forEach(this.$menuItems, function(element, index) {
 69         goog.events.listen(element, goog.events.EventType.CLICK, clickHandler, false);
 70     });
 71 };
 72 
 73 /**
 74  * Highlights the link for the current section based on it's href.
 75  * @param {string} href The target of the link to highlight.
 76  */
 77 gva.component.SectionMenu.prototype.setActive = function(href) {
 78 
 79     // If there's no slug, use default
 80     if (href === '') {
 81         var matches = goog.dom.getElementsByClass('default');
 82         if (matches.length > 0) {
 83             goog.dom.classes.add(matches[0], 'active');
 84         }
 85     } else {
 86         // Build an expression for the target to test each link against.
 87         var section = new RegExp('^\/?' + href + '\/?$');
 88         // Loop through all menu items and check their target against the goal.
 89         goog.array.forEach(this.$menuItems, function(element, index) {
 90             // If the targets match, tag the link as active.
 91             if (section.test(element.getAttribute('href'))) {
 92                 goog.dom.classes.add(element, 'active');
 93             } else {
 94                 goog.dom.classes.remove(element, 'active');
 95             }
 96         });
 97     }
 98 };
 99 
100 /**
101  * Event handler for clicked menu items.
102  * @param {goog.events.BrowserEvent} event A click event from a menu item.
103  */
104 gva.component.SectionMenu.prototype.onMenuItemClicked = function(event) {
105 
106     // Prevent default anchor behavior.
107     event.stopPropagation();
108     event.preventDefault();
109 
110     // Get the link and title from the anchor.
111     var link = event.currentTarget;
112     var href = link.getAttribute('href');
113     var title = link.getAttribute('title');
114 
115     // Navigate via ApplicationController.
116     gva.controller.ApplicationController.navigate(href, title);
117 };
