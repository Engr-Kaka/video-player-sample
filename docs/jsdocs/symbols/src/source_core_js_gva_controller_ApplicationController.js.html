 42 /**
 43  * @constructor
 44  */
 45 gva.controller.ApplicationController = (function() {
 46 
 47     /**
 48      * Whether this controller has been initialized
 49      * @type {boolean}
 50      */
 51     var _initialized = false;
 52 
 53     /**
 54      * True until the first refresh has happened
 55      * @type {boolean}
 56      */
 57     var _firstRefresh = true;
 58 
 59     /**
 60      * The current history token
 61      * @type {string}
 62      */
 63     var _currentToken = null;
 64 
 65     /**
 66      * The DOM element containing the application sections
 67      * @type {Element}
 68      */
 69     var _$sectionContainer = null;
 70 
 71     /**
 72      * The singleton instance of the global header controller
 73      * @type {gva.controller.HeaderController}
 74      */
 75     var _headerController = null;
 76 
 77     /**
 78      * The singleton instance of the global footer controller
 79      * @type {gva.controller.FooterController}
 80      */
 81     var _footerController = null;
 82 
 83     /**
 84      * The currently active section controller
 85      * @type {gva.controller.AbstractController}
 86      */
 87     var _currentController = null;
 88 
 89     /**
 90      * The key of the data load event. Used for unbind handlers
 91      */
 92     var _dataLoadEventKey = null;
 93 
 94     /**
 95      * Section controllers, indexed by slug
 96      */
 97     var _controllersBySlug = {};
 98 
 99     /**
100      * Application history handler
101      * @type {goog.history.Html5History|goog.History}
102      */
103     var _history = null;
104 
105     /**
106      * Monitors the viewport size so that the layout can resize when required
107      * @type {goog.dom.ViewportSizeMonitor}
108      */
109     var _viewportMonitor = new goog.dom.ViewportSizeMonitor();
110 
111     var _loadConfig = function() {
112 
113         this._configRequest = new goog.net.XhrIo();
114         goog.events.listenOnce(this._configRequest, goog.net.EventType.COMPLETE, _onConfigLoaded);
115         this._configRequest.send('/data/config.json?t=' + new Date().getTime());
116     };
117 
118     var _loadInitialData = function() {
119 
120         var model = gva.model.DataModel.getInstance();
121         _dataLoadEventKey = goog.events.listen(model, gva.model.DataModel.EventType.GOT_ALL_DATA, _onInitialDataLoaded, false, this);
122 
123         model.getShows();
124         model.getEpisodes();
125     };
126 
127     /**
128      * Starts up all controllers associates with sections
129      */
130     var _registerControllers = function() {
131 
132         var slug;
133         var type;
134         var element;
135         var controller;
136         var $sectionContainer = goog.dom.getElement('sections');
137         var $sections = goog.dom.getElementsByTagNameAndClass('section', null, $sectionContainer);
138 
139         /**
140          * Walk through each top level section inside the section container
141          * For each section, check whether a controller is specified and if so,
142          * start up the controller with the section as it's context
143          */
144         goog.array.forEach($sections, function(element, index) {
145 
146             // Retrieve the controller definition from the element's data attribute
147             type = goog.dom.dataset.get(element, 'controller');
148             if (!type) {
149               return;
150             }
151 
152             // Initialize this controller type with the element as it's context
153             controller = gva.controller.ControllerInitializer.initializeController(type, element);
154 
155             // Retrieve the slug which this section represents
156             slug = goog.dom.dataset.get(element, 'slug');
157 
158             // Index this controller by it's section slug
159             _controllersBySlug[slug] = controller;
160         });
161 
162         /**
163          * Initialize header controller
164          */
165         element = goog.dom.getElement('header');
166         if (element) {
167           type = goog.dom.dataset.get(element, 'controller');
168 
169           _headerController = gva.controller.ControllerInitializer.initializeController(type, element);
170         }
171 
172         /**
173          * Initialize footer controller
174          */
175         element = goog.dom.getElement('footer');
176         if (element) {
177           type = goog.dom.dataset.get(element, 'controller');
178 
179           _footerController = gva.controller.ControllerInitializer.initializeController(type, element);
180         }
181 
182         /**
183          * Initialize subscriptions controller
184          */
185         element = goog.dom.getElement('subscriptions');
186         if (element) {
187           type = goog.dom.dataset.get(element, 'controller');
188 
189           _subscriptionsController = gva.controller.ControllerInitializer.initializeController(type, element);
190         }
191     };
192 
193     /**
194      * Initializes any components in the DOM
195      */
196     var _registerComponents = function() {
197         gva.component.ComponentInitializer.registerComponents();
198     };
199 
200     /**
201      * Registers necessary event handlers
202      */
203     var _registerListeners = function() {
204 
205         // Respond to window resize events
206         goog.events.listen(_viewportMonitor, goog.events.EventType.RESIZE, function(event) {
207             if (!!_currentController) {
208                 _scrollContainerTo(_currentController.$context, 0);
209             }
210         });
211     };
212 
213     var _onConfigLoaded = function(event) {
214 
215         // Set the global config data
216         var data = goog.json.parse(event.target.getResponseText());
217 
218         // Merge the loaded config with the global config
219         goog.mixin(gva.config, data['config']);
220 
221         // Load the show data
222         _loadInitialData();
223     };
224 
225     var _onInitialDataLoaded = function(event) {
226 
227         // Unsubscribe from the loaded event
228         goog.events.unlistenByKey(_dataLoadEventKey);
229 
230         // Hide load screen.
231         var loadScreen = goog.dom.getElementByClass('load-screen');
232         if (loadScreen) {
233           fade = new goog.fx.dom.FadeOutAndHide(loadScreen, 150);
234           fade.play();
235         }
236 
237         // First register all components in the DOM
238         _registerComponents();
239 
240         // Next we can start the controllers
241         _registerControllers();
242 
243         // Try to use the history API
244         _enableHistory();
245     };
246 
247     /**
248      * Event handler for the history state change event
249      */
250     var _onHistoryNavigate = function(event) {
251         event.preventDefault();
252         var token = _history.getToken();
253 
254         if (token !== _currentToken) {
255             _currentToken = token;
256             gva.controller.ApplicationController.refresh();
257         }
258     };
259 
260     /**
261      * Enable deep linking via history API or fallback
262      */
263     var _enableHistory = function() {
264 
265         // Initialize appropriate history controller
266         if (goog.history.Html5History.isSupported()) {
267             _history = new goog.history.Html5History();
268             goog.events.listen(_history, goog.history.EventType.NAVIGATE, _onHistoryNavigate, false);
269             _history.setUseFragment(true);
270             _history.setEnabled(true);
271         }
272         else {
273           // Older browser history will require
274             $historyInput = goog.dom.getElementsByTagNameAndClass('input', 'ie-history')[0];
275             $historyIframe = goog.dom.getElementsByTagNameAndClass('iframe', 'ie-history')[0];
276             _history = new goog.History(false, null, $historyInput, $historyIframe);
277             goog.events.listen(_history, goog.history.EventType.NAVIGATE, _onHistoryNavigate, false);
278             _history.setEnabled(true);
279         }
280     };
281 
282     /**
283      * Scrolls the section container to the given coordinates
284      * @param {Element} $target The target DOM element to scroll to.
285      * @param {number=} duration The duration of the tween in milliseconds.
286      */
287     var _scrollContainerTo = function($target, duration) {
288 
289         if (isNaN(duration)) {
290             duration = 450;
291         }
292 
293         var sx = _$sectionContainer.scrollLeft;
294         var sy = _$sectionContainer.scrollTop;
295 
296         var tx = $target.offsetLeft;
297         var ty = $target.offsetTop;
298 
299         if (duration > 0) {
300             var animation = new goog.fx.dom.Scroll(_$sectionContainer, [sx, sy], [tx, ty], duration, gva.easing.quintic.easeOut);
301             animation.play();
302         } else {
303             _$sectionContainer.scrollLeft = tx;
304             _$sectionContainer.scrollTop = ty;
305         }
306     };
307 
308     /**
309      * Public API
310      */
311     return {
312 
313         /**
314          * Initializes the controller and starts necessary sub controllers
315          */
316         initialize: function() {
317 
318             if (!_initialized) {
319 
320                 _$sectionContainer = goog.dom.getElement('sections');
321 
322                 // Register event handlers
323                 _registerListeners();
324 
325                 // Load the application config first
326                 _loadConfig();
327 
328                 // Only do all this once!
329                 _initialized = true;
330             }
331         },
332 
333         /**
334          * Navigates to a given path pushes the state into the browser history
335          * @param {string} path The full path to navigate to.
336          * @param {string=} title An optional page title for the path.
337          * @param {Object=} state An optional state object to store against this history item.
338          */
339         navigate: function(path, title, state) {
340             _history.setToken(path, title);
341         },
342 
343         /**
344          * Returns an array containing each segment of the current URL or hash bang
345          * @return {Array.<string>} The path segments of the current URL or hash bang.
346          */
347         getPathSegments: function() {
348 
349             // Retrieve the current history token and strip extra slashes
350             var token = _history.getToken().replace(/^\/|\/$/g, '');
351             var segments = token.split('/');
352 
353             return segments;
354         },
355 
356         /**
357          * Refreshes the application state based on the current history token
358          */
359         refresh: function() {
360 
361             // Determine the current path
362             var segments = gva.controller.ApplicationController.getPathSegments();
363 
364             // Retrieve the controller associated with this section
365             var controller = _controllersBySlug[segments[0]];
366 
367             // If the controller exists
368             if (typeof(controller) !== 'undefined') {
369 
370                 // Disable the current controller if different
371                 if (_currentController && _currentController !== controller) {
372                     _currentController.disable();
373                 }
374 
375                 // Are we changing controller or not
376                 var isCurrent = _currentController === controller;
377 
378                 // Activate the section controller
379                 _currentController = controller;
380                 _currentController.enable();
381 
382                 // Scroll to the current section
383                 _scrollContainerTo(_currentController.$context, _firstRefresh ? 0 : 450);
384 
385                 // Refresh the footer
386                 _footerController.refresh();
387 
388                 // Any subsections are the responsibility of the section controller
389                 if (typeof(_currentController.navigate) === 'function') {
390 
391                     // Clone the segments array and strip out root segment
392                     var subsections = segments.concat();
393                     subsections.shift();
394 
395                     // Tell the section controller to handle subsections
396                     _currentController.navigate(subsections, _firstRefresh || !isCurrent);
397                 }
398             } else if (segments[0] === '') {
399 
400                 // Refresh the footer on the home page
401                 if (_footerController) {
402                   _footerController.refresh();
403                 }
404             }
405 
406             _firstRefresh = false;
407         }
408     };
