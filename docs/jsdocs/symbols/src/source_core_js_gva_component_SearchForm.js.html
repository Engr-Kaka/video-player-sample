 37 /**
 38  * @constructor
 39  * @extends {gva.component.BaseComponent}
 40  */
 41 gva.component.SearchForm = function() {
 42 
 43     this._$searchInput = null;
 44     this._$searchReset = null;
 45     this._$selected = null;
 46     this._$results = null;
 47     this._$info = null;
 48     this._placeholder = null;
 49     this._lastQuery = null;
 50     this._callbacks = null;
 51 
 52     // The height of each result item.
 53     this.RESULT_LI_HEIGHT = 101;
 54 
 55     // The height of the num results info and padding.
 56     this.RESULT_BASE_HEIGHT = 52;
 57 
 58     // The difference between the base height and the base offsetHeight
 59     this.RESULT_BASE_OFFSET = 13;
 60 
 61     // Additional height from 12px margin added when 1 or more results found.
 62     this.RESULT_BOTTOM_MARGIN = 12;
 63 
 64     // Additional 40px 'more' link added when more than 6 results are found.
 65     this.RESULT_MORE_HEIGHT = 40;
 66 
 67     // Width of results drop down.
 68     this.RESULT_WIDTH = 319;
 69 
 70     this.RESULT_ANIMATION_DURATION = 250;
 71 
 72     // Maximum number of results to display in dropdown results.
 73     this.MAX_LIVE_RESULTS = 6;
 74 
 75     // Call super constructor
 76     gva.component.BaseComponent.call(this);
 77 };

 86 /**
 87  * @inheritDoc
 88  */
 89 gva.component.SearchForm.prototype.initialize = function($context) {
 90 
 91     // Call super method.
 92     this.constructor.superClass_.initialize.call(this, $context);
 93 
 94     this.callbacks = {
 95             onKeyUp: goog.bind(this.onKeyUp, this),
 96             onInputChange: goog.bind(this.onInputChange, this),
 97             onFormSubmit: goog.bind(this.onFormSubmit, this),
 98             onSearchFocus: goog.bind(this.onSearchFocus, this),
 99             onSearchBlur: goog.bind(this.onSearchBlur, this),
100             onEpisodeClicked: goog.bind(this.onEpisodeClicked, this),
101             onResetClicked: goog.bind(this.onResetClicked, this),
102             onShowClicked: goog.bind(this.onShowClicked, this),
103             handleClick: goog.bind(this.handleClick, this)
104     };
105 
106     // Grab the search input element
107     this._$searchInput = goog.dom.query('input[name="search"]', this.$context)[0];
108 
109     // Grab the reset button
110     this._$searchReset = goog.dom.query('input[type="reset"]', this.$context)[0];
111 
112     // Listen for arrow key events on form
113     goog.events.listen(this.$context, goog.events.EventType.KEYUP, this.callbacks.onKeyUp, false);
114 
115     // Listen the form events
116     goog.events.listen(this._$searchInput, goog.events.EventType.KEYUP, this.callbacks.onInputChange, false);
117     goog.events.listen(this.$context, goog.events.EventType.SUBMIT, this.callbacks.onFormSubmit, false);
118 
119     // Listen for focus / blur events for setting place holder text
120     goog.events.listen(this._$searchInput, goog.events.EventType.FOCUSIN, this.callbacks.onSearchFocus, false);
121     goog.events.listen(this._$searchInput, goog.events.EventType.FOCUSOUT, this.callbacks.onSearchBlur, false);
122 
123     // Listen for clicks on the reset button
124     goog.events.listen(this._$searchReset, goog.events.EventType.CLICK, this.callbacks.onResetClicked, false);
125 
126     // Use the initial value of the search input as the place holder
127     this._placeholder = goog.dom.forms.getValue(this._$searchInput);
128 
129     // Build the results container
130     this.initResults();
131 };

134 /**
135  * Builds the drop down results container.

137 gva.component.SearchForm.prototype.initResults = function() {

143     goog.dom.appendChild(this.$context, this._$results);

149 /**
150  * Creates a DOM element for an episode result list item.
151  * @param {gva.vo.Episode} episode The data for the result item.
152  * @return {Element} The element for the episode's entry in the results list.
153  */
154 gva.component.SearchForm.prototype.makeResult = function(episode) {
155 
156     var showSlug = gva.util.StringUtil.slugify(episode.show_title);
157     var episodeSlug = gva.util.StringUtil.slugify(episode.title);
158     var episodeText = gva.util.StringUtil.stripTags(episode.text);
159     episodeText = gva.util.StringUtil.singleSentence(episodeText, 15, 60);
160 
161     var markup = '' +
162         '<li>' +
163             '<div>' +
164             '<a class="episode" href="/home/' + episode.slug + '" title="' + episode.title + '" data-episodeid="' + episode.id + '">' +
165                 '<img src="' + episode.image_url + '"/>' +
166                 '<strong>' + episode.title + '</strong>' +
167                 '<em>' + episodeText + '</em>' +
168              '</a>' +
169              '<a class="show" href="/shows/' + showSlug + '/" title="' + episode.show_title + '">' + episode.show_title + '</a>' +
170              '</div>' +
171         '</li>';
172 
173     return goog.dom.htmlToDocumentFragment(markup);
174 };

177 /**
178  * Empties episode results from drop down.
179  */
180 gva.component.SearchForm.prototype.clearResults = function() {
181 
182     var $children = goog.dom.getChildren(this._$results);
183     var callbacks = this.callbacks;
184 
185     // Remove children and event handlers
186     goog.array.forEach($children, function($element) {
187 
188         $episodeLink = goog.dom.query('a.episode', $element)[0];
189         $showLink = goog.dom.query('a.show', $element)[0];
190 
191         if (typeof($episodeLink) !== 'undefined') {
192             goog.events.unlisten($episodeLink, goog.events.EventType.CLICK, callbacks.onEpisodeClicked);
193             goog.events.unlisten($showLink, goog.events.EventType.CLICK, callbacks.onShowClicked);
194         }
195     });
196 
197     goog.dom.removeChildren(this._$results);
198 };

201 /**
202  * Event handler for when the search input is focused.
203  * @param {goog.events.Event} event The event which triggered the handler.
204  */
205 gva.component.SearchForm.prototype.onSearchFocus = function(event) {
206 
207     var value = goog.dom.forms.getValue(this._$searchInput);
208 
209     if (value === this._placeholder) {
210         goog.dom.forms.setValue(this._$searchInput, '');
211         goog.dom.classes.remove(this._$searchInput, 'placeholder');
212     }
213 };

216 /**
217  * Event handler for when the search input is blurred.
218  * @param {goog.events.Event} event The event which triggered the handler.
219  */
220 gva.component.SearchForm.prototype.onSearchBlur = function(event) {
221 
222     var value = goog.dom.forms.getValue(this._$searchInput);
223 
224     if (value.replace(/\s+/g, '').length === 0) {
225         goog.dom.forms.setValue(this._$searchInput, 'Search');
226         goog.dom.classes.add(this._$searchInput, 'placeholder');
227     }
228 };

231 /**
232  * Event handler for keyUp events while typing in the search input.
233  * @param {goog.events.Event} event The event which triggered the handler.
234  */
235 gva.component.SearchForm.prototype.onKeyUp = function(event) {
236 
237     event.preventDefault();
238 
239     switch (event.keyCode) {
240 
241         case goog.events.KeyCodes.UP:
242 
243             // If no results are selected, focus the first result
244             if (this._$selected) {
245 
246                 // Down go up further than the first result
247                 if (this._$selected.previousSibling === this._$info) {
248                     return;
249                 }
250 
251                 // Select the previous result
252                 if (this._$selected.previousSibling) {
253                     this._$selected = this._$selected.previousSibling;
254                 }
255             }
256 
257             // Focus the first link inside the result
258             goog.dom.query('a', this._$selected)[0].focus();
259 
260             break;
261 
262         case goog.events.KeyCodes.DOWN:
263 
264             // If no results are selected, focus the first result
265             if (!this._$selected) {
266                 this._$selected = goog.dom.getChildren(this._$results)[1];
267             } else {
268 
269                 // Select the next result
270                 if (this._$selected.nextSibling) {
271                     this._$selected = this._$selected.nextSibling;
272                 }
273             }
274 
275             // Focus the first link inside the result
276             goog.dom.query('a', this._$selected)[0].focus();
277 
278             break;
279     }
280 };

283 /**
284  * Event handler for when the search input value changes.
285  * @param {goog.events.Event} event The event which triggered the handler.
286  */
287 gva.component.SearchForm.prototype.onInputChange = function(event) {
288 
289     var query = goog.dom.forms.getValue(this._$searchInput);
290 
291     // Show / hide the reset toggle
292     goog.style.setStyle(this._$searchReset, 'opacity', query === '' ? '0' : '1');
293 
294     if (query.replace(/\s+/g, '').length < 1) {
295         goog.style.showElement(this._$results, false);
296         return;
297     }
298 
299     if (this._lastQuery && query === this._lastQuery) {
300         return;
301     }
302 
303     this._lastQuery = query;
304 
305     var model = gva.model.DataModel.getInstance();
306     var results = model.searchEpisodes(query);
307 
308     this.clearResults();
309     // Remove current results
310     //goog.dom.removeChildren(this._$results);
311 
312     // Append results heading
313     goog.dom.append(this._$results, this._$info);
314     goog.dom.setTextContent(this._$info, results.length + ' Results');
315 
316     // Add some results
317 
318     var $result;
319     var $showLink;
320     var $episodeLink;
321     var count = Math.min(this.MAX_LIVE_RESULTS, results.length);
322 
323     for (var i = 0; i < count; i++) {
324         $result = this.makeResult(results[i]);
325 
326         // Initialize buttons
327         $episodeLink = goog.dom.query('a.episode', $result)[0];
328         $showLink = goog.dom.query('a.show', $result)[0];
329 
330         // Bind event handlers to the object
331         goog.events.listen($episodeLink, goog.events.EventType.CLICK, this.callbacks.onEpisodeClicked);
332         goog.events.listen($showLink, goog.events.EventType.CLICK, this.callbacks.onShowClicked);
333 
334         goog.dom.append(this._$results, $result);
335     }
336 
337     if (results.length > 6) {
338       // Add "more" link to results page.
339       var $more = goog.dom.htmlToDocumentFragment('<li class="more"><a href="#">More...</a></li>');
340       goog.events.listen($more, goog.events.EventType.CLICK, this.callbacks.onFormSubmit);
341 
342       goog.dom.append(this._$results, $more);
343     }
344 
345     // Reveal results.
346     var startHeight = this._$results.offsetHeight - this.RESULT_BASE_OFFSET;
347 
348     // Make visible before animation.
349     var initialReveal = !goog.style.isElementShown(this._$results);
350     goog.style.showElement(this._$results, true);
351 
352 
353     // Calculate the ending height for the result animation.
354     var endHeight = count * this.RESULT_LI_HEIGHT;
355     endHeight += (endHeight > 0) ? this.RESULT_BOTTOM_MARGIN : 0;
356     endHeight += (results.length > 6) ? this.RESULT_MORE_HEIGHT : 0;
357     endHeight += this.RESULT_BASE_HEIGHT;
358 
359 
360     // Animate height change if different.
361     if (startHeight != endHeight) {
362 
363       // Hide inner list elements hidden before animating.
364       this.toggleResultDisplay(false, !initialReveal);
365 
366       var anim = new goog.fx.dom.Resize(this._$results,
367         [this.RESULT_WIDTH, startHeight],
368         [this.RESULT_WIDTH, endHeight],
369         this.RESULT_ANIMATION_DURATION,
370         gva.easing.quintic.easeOut
371       );
372 
373       goog.events.listen(anim, goog.fx.Transition.EventType.END, function() {
374         this.toggleResultDisplay(true);
375       }, false, this);
376 
377       anim.play();
378     }
379 
380 
381     // Listen for closing clicks.
382     goog.events.unlisten(window, goog.events.EventType.CLICK, this.callbacks.handleClick, false);
383     goog.events.listen(window, goog.events.EventType.CLICK, this.callbacks.handleClick, false);
384 
385     // Deselect results
386     this._$selected = null;
387 };

389 /**
390  * Toggles the display of each li element in the results list to block or none.
391  * @param {Boolean} display True to display block, false for none.
392  * @param {Boolean} skipIntro True to ignore the first li (which shows the num results).
393  */
394 gva.component.SearchForm.prototype.toggleResultDisplay = function(display, skipIntro) {
395 
396   var displayStyle = display ? 'block' : 'none';
397 
398   var $resultItems = goog.dom.getElementsByTagNameAndClass('li', null, this._$results);
399 
400   for (var i = 0, n = $resultItems.length; i < n; i++) {
401 
402     if (i == 0 && skipIntro) {
403       continue;
404     }
405 
406     goog.style.setStyle($resultItems[i], 'display', displayStyle);
407   }
408 };

410 /**

414 gva.component.SearchForm.prototype.onResetClicked = function(event) {
415     event.preventDefault();

452 /**
453  * Event handler for when the search input is submitted.
454  * @param {goog.events.Event} event The event which triggered the handler.
455  */
456 gva.component.SearchForm.prototype.onFormSubmit = function(event) {
457 
458     // Stop default form behavior
459     event.preventDefault();
460 
461     // Grab the query from the search field
462     var query = goog.dom.forms.getValue(this._$searchInput);
463 
464     // Close results view.
465     goog.style.showElement(this._$results, false);
466 
467     // Go to the search results page
468     gva.controller.ApplicationController.navigate('/search/' + query);
469 };

472 /**
473  * Event handler for all clicks that collapses the component when clicking outside of it.
474  * @param {goog.events.Event} event The event which triggered the handler.
475  */
476 gva.component.SearchForm.prototype.handleClick = function(event) {
477   var resultsAncestor = goog.dom.getAncestorByClass(event.target, 'results');
478   var isLink = goog.dom.classes.has(event.target, 'episode') || goog.dom.classes.has(event.target.parentNode, 'episode') || goog.dom.classes.has(event.target, 'show');
479 
480   if (!resultsAncestor || isLink) {
481     // Close results view.
482     goog.style.showElement(this._$results, false);
483 
484     goog.events.unlisten(window, goog.events.EventType.CLICK, this.callbacks.handleClick, false);
485   }
486 };
