 32 /**
 33  * @constructor
 34  * @extends {gva.component.BaseComponent}
 35  */
 36 gva.component.Html5Player = function() {
 37 
 38     // Indicates when the user is seeking.
 39     this.isSeeking = false;
 40 
 41     // The progress bar.
 42     this.seekSlider = null;
 43 
 44     /**
 45      * Monitors the viewport size so that the grid can flex.
 46      * @type {goog.dom.ViewportSizeMonitor}
 47      */
 48     this.viewportMonitor = new goog.dom.ViewportSizeMonitor();
 49 
 50     // Whether or not to resume after seeking.
 51     this.wasPlaying = false;
 52 
 53     /**
 54      * Enumeration for player event types.
 55      * @enum {string}
 56      */
 57     gva.component.Html5Player.EventType = {
 58             EPISODE_COMPLETE: 'Html5Player::episodeComplete'
 59     };
 60 
 61     // Call super constructor
 62     gva.component.BaseComponent.call(this);
 63 };

 88 /**
 89  * Creates the markup for the video player and caches DOM
 90  * queries for parts of the video player such as controls.
 91  */
 92 gva.component.Html5Player.prototype.createBase = function() {
 93 
 94   // Markup for video and controls.
 95 
 96   var html = '<div id="html5Player">' +
 97       '<div class="video-container"><video autoplay preload="auto"></video></div>' +
 98       '<div class="video-controls">' +
 99         '<div class="video-controls-wrap">' +
100           '<div class="video-play video-button"><span></span></div>' +
101           '<div class="video-seek">' +
102             '<div class="video-seek-wrap goog-slider-horizontal">' +
103               '<span class="bar-bg"></span>' +
104               '<span class="bar-fill"></span>' +
105               '<span class="bar-knob goog-slider-thumb">' +
106                 '<span class="knob-graphic"></span>' +
107               '</span>' +
108             '</div>' +
109           '</div>' +
110           '<div class="video-sound">' +
111             '<div class="video-sound-button video-button"><span></span></div>' +
112             '<div class="video-sound-panel">' +
113               '<div class="video-sound-panel-wrap goog-slider-horizontal">' +
114                 '<span class="bar-bg"></span>' +
115                 '<span class="bar-fill"></span>' +
116                 '<span class="bar-knob goog-slider-thumb">' +
117                   '<span class="knob-graphic"></span>' +
118                 '</span>' +
119               '</div>' +
120             '</div>' +
121           '</div>' +
122           '<div class="video-time">' +
123             '<span class="time-current">00:00</span>' +
124             '<span class="time-total">00:00</span>' +
125           '</div>' +
126         '</div>' +
127       '</div>' +
128     '</div>';
129 
130   this.root = goog.dom.htmlToDocumentFragment(html);
131   goog.dom.appendChild(this.$context, this.root);
132 
133   // Cache player elements.
134 
135   this.player = goog.dom.getElementByClass('video-container', this.root);
136   this.video = goog.dom.getFirstElementChild(this.player);
137 
138   this.controls = goog.dom.getElementByClass('video-controls', this.root);
139   this.playBtn = goog.dom.getElementByClass('video-play', this.controls);
140 
141   this.seekPanel = goog.dom.getElementByClass('video-seek-wrap', this.controls);
142   this.seekFill = goog.dom.getElementByClass('bar-fill', this.seekPanel);
143   this.seekKnob = goog.dom.getElementByClass('bar-knob', this.seekPanel);
144 
145   this.timeCurrent = goog.dom.getElementByClass('time-current', this.controls);
146   this.timeTotal = goog.dom.getElementByClass('time-total', this.controls);
147 
148   this.soundBtn = goog.dom.getElementByClass('video-sound-button', this.controls);
149   this.soundPanel = goog.dom.getElementByClass('video-sound-panel-wrap', this.controls);
150   this.soundFill = goog.dom.getElementByClass('bar-fill', this.soundPanel);
151   this.soundKnob = goog.dom.getElementByClass('bar-knob', this.soundPanel);
152 
153 
154   // Initial event listeners
155 
156   // Lock default select behavior.
157   goog.events.listen(this.controls, goog.events.EventType.SELECTSTART, function() {
158     return false;
159   });
160 
161   // Listen for video progress and update the seek slider position.
162   goog.events.listen(this.video, 'timeupdate', this.onVideoUpdate, false, this);
163 
164 
165   // Listen for play/pause toggle clicks.
166   goog.events.listen(this.playBtn, goog.events.EventType.CLICK, this.play, false, this);
167 
168   // Listen for volume button clicks to toggle mute.
169   goog.events.listen(this.soundBtn, goog.events.EventType.CLICK, this.muteToggle, false, this);
170 
171   // Listen for end of video.
172   goog.events.listen(this.video, 'ended', this.onEnded, false, this);
173 
174   // Listen for resize events.
175   goog.events.listen(this.viewportMonitor, goog.events.EventType.RESIZE, this.resize, false, this);
176 };

180 /**
181  * Handler for when the video's ready to play.
182  */
183 gva.component.Html5Player.prototype.onCanPlay = function() {
184 
185   // Set initial time display.
186   this.timeTotal.innerHTML = this.timeFormat(this.video.duration);
187 
188   // Show controls.
189   goog.style.setStyle(this.controls, 'display', 'block');
190 
191   // Activate seek/volume sliders.
192   this.createSliders();
193 
194   // Set initial volume.
195   this.setVolume(this.video.volume * 100);
196 
197   // Set play button state.
198   goog.dom.classes.add(this.playBtn, 'playing');
199 
200   // Set sizing for IE to avoid flickering when controls are on top of the video.
201   if (goog.userAgent.IE) {
202     this.IEResizeFix();
203   }
204 };

207 /**
208  * Creates and sets event listeners for
209  * seek and volume sliders
210  */
211 gva.component.Html5Player.prototype.createSliders = function() {
212 
213   // Create and bind seek slider.
214   this.seekSlider = new goog.ui.Slider;
215   this.seekSlider.decorate(this.seekPanel);
216   this.seekSlider.setMaximum(this.video.duration);
217   this.seekSlider.setMoveToPointEnabled(true);
218   goog.events.listen(this.seekSlider,
219     goog.events.EventType.CHANGE, this.onSeekChange,
220     false, this);
221 
222 
223   // Create and bind volume slider.
224   this.volumeSlider = new goog.ui.Slider;
225   this.volumeSlider.decorate(this.soundPanel);
226   goog.events.listen(this.volumeSlider,
227     goog.events.EventType.CHANGE, this.onVolumeChange,
228     false, this);
229 
230 
231   // Listen for dragging on the seek slider.
232   goog.events.listen(this.seekPanel,
233     goog.events.EventType.MOUSEDOWN, this.onSeekStart,
234     false, this);
235 };

238 /**
239  * Sets the src attribute loading the video.
240  * @param {String} url The url for the video to be loaded.
241  */
242 gva.component.Html5Player.prototype.loadVideo = function(url) {
243 
244   // Load a null video to help convince browsers to free up memory.
245   this.video.src = '';
246   this.video.load();
247 
248   // Load the video by url.
249   this.video.src = url;
250   this.video.load();
251 
252 
253   // Call other setup functions when ready.
254   if (this.video.readyState > this.video.HAVE_CURRENT_DATA) {
255     this.onCanPlay();
256   }
257   else {
258     goog.events.listen(this.video,
259       'canplay', this.onCanPlay,
260       false, this);
261   }
262 
263 };

266 /**
267  * Toggles play/pause for the video.
268  */
269 gva.component.Html5Player.prototype.play = function() {
270 
271   if (!this.video.src) {
272     return;
273   }
274 
275   if (this.video.paused) {
276     this.video.play();
277     goog.dom.classes.add(this.playBtn, 'playing');
278   }
279 
280   else if (this.video.ended) {
281     this.video.currentTime = 0;
282     this.video.play();
283     goog.dom.classes.add(this.playBtn, 'playing');
284   }
285 
286   else {
287     this.video.pause();
288     goog.dom.classes.remove(this.playBtn, 'playing');
289   }
290 };

314 /**
315  * Toggles mute for the video.
316  */
317 gva.component.Html5Player.prototype.muteToggle = function() {
318   if (this.video.muted) {
319     this.video.muted = false;
320     goog.dom.classes.remove(this.soundBtn, 'sound-off');
321 
322     // Bump up volume if set to 0.
323     if (this.volumeSlider.getValue() == 0) {
324       this.setVolume(50);
325     }
326   }
327   else {
328     this.setVolume(0);
329     this.video.muted = true;
330     goog.dom.classes.add(this.soundBtn, 'sound-off');
331   }
332 };

344 /**
345  * Handler for when the video progress updates.
346  * Updates the slider and numerical time displays.
347  */
348 gva.component.Html5Player.prototype.onVideoUpdate = function() {
349 
350   // Update displayed time.
351   this.timeCurrent.innerHTML = this.timeFormat(this.video.currentTime);
352 
353 
354   // Don't update while scrubbing since the slider is already being
355   // moved to the desired position.
356 
357   if (!this.isSeeking) {
358     this.seekSlider.setValue(this.video.currentTime);
359   }
360 };

363 /**
364  * Handles mousedown (dragging) events for the seek slider.
365  */
366 gva.component.Html5Player.prototype.onSeekStart = function() {
367 
368   if (!this.video.paused && !this.video.ended) {
369     this.wasPlaying = true;
370   }
371   else {
372     this.wasPlaying = false;
373   }
374   this.pause();
375 
376   // Unbind scrub start listener.
377   goog.events.unlisten(this.seekPanel, goog.events.EventType.MOUSEDOWN, this.onSeekStart, false, this);
378 
379   // Listen for scrub stop.
380   goog.events.listenOnce(document, goog.events.EventType.MOUSEUP, this.onSeekStop, false, this);
381 
382   // Set conditional so only user can reposition
383   // the seek slider while scrubbing.
384   this.isSeeking = true;
385 
386 
387   // Update seek position once right away for single clicks
388   // on the slider rather than scrubs.
389   this.onSeekChange();
390 };
391 
392 
393 /**
394  * Handles mouseup events after seeking has stopped.
395  */
396 gva.component.Html5Player.prototype.onSeekStop = function() {
397 
398   // Listen for seeking again.
399   goog.events.listen(this.seekPanel,
400     goog.events.EventType.MOUSEDOWN,
401     this.onSeekStart,
402     false, this);
403 
404   // Allow seek slider to be programmatically updated again.
405   this.isSeeking = false;
406 
407   if (this.wasPlaying) {
408     this.play();
409   }
410 };

413 /**
414  * Handles changes to the video seek slider.
415  */
416 gva.component.Html5Player.prototype.onSeekChange = function() {
417 
418   // Update video time position.
419   if (this.isSeeking) {
420     this.video.currentTime = this.seekSlider.getValue();
421   }
422 
423   // Keep the slider's fill bar pinned to the knob.
424   var knobOffset = goog.style.getStyle(this.seekKnob, 'left');
425   goog.style.setStyle(this.seekFill, 'width', knobOffset);
426 };
427 
428 
429 /**
430  * Handles changes to the sound panel slider.
431  */
432 gva.component.Html5Player.prototype.onVolumeChange = function() {
433 
434   // Make sure it's not muted.
435   if (this.video.muted) {
436     this.muteToggle();
437   }
438 
439   // Update video volume.
440   this.video.volume = this.volumeSlider.getValue() / 100;
441 
442   // Keep the slider's fill bar pinned to the knob.
443   var knobOffset = goog.style.getStyle(this.soundKnob, 'left');
444   goog.style.setStyle(this.soundFill, 'width', knobOffset);
445 };

448 /**
449  * Handles browser resize events to scale and reorient player elements.
450  */
451 gva.component.Html5Player.prototype.resize = function() {
452   if (goog.userAgent.IE) {
453     this.IEResizeFix();
454   }
455 
456   // Refresh seek knob position.
457   if (this.seekSlider) {
458     this.seekSlider.animatedSetValue(this.seekSlider.getValue());
459     goog.Timer.callOnce(this.onSeekChange, 200, this);
460   }
461 };

464 /**
465  * Resizes video element so controls are not on top to prevent flickering in IE.
466  */
467 gva.component.Html5Player.prototype.IEResizeFix = function() {
468   var availableHeight = goog.style.getPosition(this.controls).y;
469   goog.style.setStyle(this.video, 'height', availableHeight + 'px');
470 };

473 /**
474  * Formats time as MM:SS.
475  * @param {Number} seconds Integer time (s) to format.
476  * @return {String} formattedTime containing formatted time.
477  */
478 gva.component.Html5Player.prototype.timeFormat = function(seconds) {
479 
480   // Calculate minutes and seconds.
481   var m = Math.floor(seconds / 60);
482   var s = Math.floor(seconds % 60);
483 
484   // Format with leading zeroes.
485   var formattedTime = (m < 10) ? '0' + m : m;
486   formattedTime += ':';
487   formattedTime += (s < 10) ? '0' + s : s;
488 
489   return formattedTime;
490 };
