 35 /**
 36  * @constructor
 37  * @extends {gva.component.BaseComponent}
 38  */
 39 gva.component.ShowDetails = function() {
 40 
 41     this.currentHeight = 0;
 42 
 43     this._$tabs = null;
 44     this._$sections = null;
 45     this._$subscribe = null;
 46     this._$about = null;
 47     this._showData = null;
 48     this._callbacks = null;
 49 
 50     // Call super constructor
 51     gva.component.BaseComponent.call(this);
 52 };
 53 
 54 /**
 55  * Extend gva.component.BaseComponent
 56  */
 57 goog.inherits(gva.component.ShowDetails, gva.component.BaseComponent);
 58 
 59 /**
 60  * Enumeration for this component's event types
 61  */
 62 gva.component.ShowDetails.EventType = {
 63         RESIZE: 'ShowDetails::resize'
 64 };
 65 
 66 /**
 67  * @inheritDoc
 68  */
 69 gva.component.ShowDetails.prototype.initialize = function($context) {
 70 
 71     // Call super method.
 72     this.constructor.superClass_.initialize.call(this, $context);
 73 
 74     // Query the DOM for the items we need
 75     this._$tabs = goog.dom.query('nav a[class~="tab"]', $context);
 76     this._$sections = goog.dom.query('section', $context);
 77     this._$subscribe = goog.dom.query('nav a.subscribe', $context)[0];
 78     this._$about = goog.dom.query('section.about article', $context)[0];
 79 
 80     // Create scoped callbacks for event handlers
 81     var callbacks = this._callbacks = {
 82             onSubscribeClicked: goog.bind(this.onSubscribeClicked, this),
 83             onTabClicked: goog.bind(this.onTabClicked, this)
 84     };
 85 
 86     // Bind subscribe click handler
 87     goog.events.listen(this._$subscribe, goog.events.EventType.CLICK, callbacks.onSubscribeClicked, false);
 88 
 89     // Bind tab click handlers
 90     goog.array.forEach(this._$tabs, function($item) {
 91         goog.events.listen($item, goog.events.EventType.CLICK, callbacks.onTabClicked, false);
 92     });
 93 };
 94 
 95 /**
 96  * Refreshes the view using the currently active show data
 97  */
 98 gva.component.ShowDetails.prototype.refresh = function() {
 99 
100     // If a show has been set, update the view
101     if (this._showData) {
102         if (this._showData.subscribed) {
103             goog.dom.classes.add(this._$subscribe, 'subscribed');
104             goog.dom.setTextContent(this._$subscribe, 'Remove');
105         } else {
106             goog.dom.classes.remove(this._$subscribe, 'subscribed');
107             goog.dom.setTextContent(this._$subscribe, 'Subscribe');
108         }
109     }
110 };
111 
112 /**
113  * Populates the component from a show data object
114  * @param {gva.vo.Show} show The show object to populate the component from.
115  */
116 gva.component.ShowDetails.prototype.setShow = function(show) {
117 
118     // Keep a reference to the current show data object
119     this._showData = show;
120 
121     var showMarkup = '' +
122         '<img src="' + show.image_url + '"/>' +
123         '<div class="description">' + show.description + '</div>';
124 
125     var $content = goog.dom.htmlToDocumentFragment(showMarkup);
126 
127     goog.dom.setTextContent(this._$about, '');
128     goog.dom.appendChild(this._$about, $content);
129 
130     // Call refresh in order to update the view for the current show obeject
131     this.refresh();
132 };

135 /**
136  * Event handler for when the subscribe button is clicked.
137  * @param {goog.events.Event} event The event which triggered the handler.
138  */
139 gva.component.ShowDetails.prototype.onSubscribeClicked = function(event) {
140 
141     event.preventDefault();
142 
143     var $item = event.currentTarget;
144     var subscribed = goog.dom.classes.has($item, 'subscribed');
145 
146     // Add or remove the subscription based on the current state
147     if (subscribed) {
148         gva.model.UserModel.getInstance().removeUserSubscription(this._showData.id);
149     } else {
150         gva.model.UserModel.getInstance().addUserSubscription(this._showData.id);
151     }
152 };

155 /**
156  * Event handler for when either of the drop down tabs are clicked.
157  * @param {goog.events.Event} event The event which triggered the handler.
158  */
159 gva.component.ShowDetails.prototype.onTabClicked = function(event) {
160 
161     event.preventDefault();
162 
163     var $tab = event.currentTarget;
164     var href = $tab.getAttribute('href');
165     var sectionClass = href.replace(/^#/, '');
166 
167     // Show / hide tabs.
168     goog.array.forEach(this._$tabs, function($item) {
169 
170         var item_href = $item.getAttribute('href');
171 
172         if ($item === $tab) {
173             goog.dom.classes.toggle($item, 'active');
174         } else {
175             goog.dom.classes.remove($item, 'active');
176         }
177     });
178 
179     // Expand / contract sections
180     goog.array.forEach(this._$sections, function($item) {
181         if (goog.dom.classes.has($item, sectionClass)) {
182             // Tag the item as expanded
183             goog.dom.classes.toggle($item, 'expanded');
184 
185         } else {
186             // Mark container contracted and close it
187             goog.dom.classes.remove($item, 'expanded');
188         }
189         goog.style.setStyle($item, 'height', '0px');
190     });
191 
192     // Check whether a section has been expanded
193     var $expanded = goog.dom.query('section[class~="expanded"]', this.$context)[0];
194 
195     // Update the current height
196     this.currentHeight = 0;
197 
198     // If is has, resize to it's height, otherwise resize back to 0
199     if ($expanded) {
200         // Compute the content height based on the first child
201         var children = goog.dom.getChildren($expanded);
202         if (!!children.length) {
203             var $content = children[0];
204             var bounds = goog.style.getBounds($content);
205             this.currentHeight = bounds.height;
206             goog.style.setStyle($expanded, 'height', this.currentHeight + 'px');
207         }
208     }
209 
210     // Dispatch the resize event
211     this.dispatchEvent(gva.component.ShowDetails.EventType.RESIZE);
212 };
