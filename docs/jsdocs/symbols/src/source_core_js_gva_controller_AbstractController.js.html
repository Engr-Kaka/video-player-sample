 17 /**
 18  * Provide a namespace for this class
 19  */
 20 goog.provide('gva.controller.AbstractController');
 21 
 22 /**
 23  * Creates a new AbstractController instance
 24  * @constructor
 25  */
 26 gva.controller.AbstractController = function() {
 27     this.$context = null;
 28     this.enabled = false;
 29     this.initialized = false;
 30 };
 31 
 32 /**
 33  * Initializes the controller with a given context
 34  * @param {Element} $context The context or view for this controller.
 35  */
 36 gva.controller.AbstractController.prototype.initialize = function($context) {
 37     if (!this.initialized) {
 38         this.initialized = true;
 39         this.$context = $context;
 40     }
 41 };
 42 
 43 /**
 44  * Marks the controller as active and starts up any sub processes
 45  */
 46 gva.controller.AbstractController.prototype.enable = function() {
 47     this.enabled = true;
 48 };
 49 
 50 /**
 51  * Marks the controller as inactive and pauses / stops any sub processes
 52  */
 53 gva.controller.AbstractController.prototype.disable = function() {
 54     this.enabled = false;
 55 };
