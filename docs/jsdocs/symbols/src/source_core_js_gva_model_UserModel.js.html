 81 /**
 82  * Retrieves a token from local storage and returns the parsed data
 83  * @param {string} token The name of the token to retrieve.
 84  * @return {Array} The data stored under the given token.
 85  */
 86 gva.model.UserModel.prototype.getDataAtToken = function(token) {
 87 
 88     // Only proceed if we have access to local storage
 89     if (!Modernizr.localstorage) { return null; }
 90 
 91     var data;
 92 
 93     // If the item doesn't yet exist, create it
 94     if (!localStorage.getItem(token)) {
 95         data = [];
 96         localStorage.setItem(token, goog.json.serialize(data));
 97     }
 98 
 99     // Return the parsed storage item
100     data = localStorage.getItem(token);
101     return goog.json.parse(data);
102 };
103 
104 /**
105  * Pushes data into a given token
106  * @param {string} token The token to add data to.
107  * @param {Object} data The data to add to the given token.
108  */
109 gva.model.UserModel.prototype.addValueToToken = function(token, data) {
110 
111     // Only proceed if we have access to local storage
112     if (!Modernizr.localstorage) { return; }
113 
114     var current = this.getDataAtToken(token);
115 
116     if (!goog.array.contains(current, data)) {
117         current.push(data);
118     }
119 
120     // Return the modified data to local storage
121     localStorage.setItem(token, goog.json.serialize(current));
122 };
123 
124 /**
125  * Removes a value from a given token
126  * @param {string} token The token to remove data from.
127  * @param {Object} data The item to remove from the given token.
128  */
129 gva.model.UserModel.prototype.removeValueFromToken = function(token, data) {
130 
131     // Only proceed if we have access to local storage
132     if (!Modernizr.localstorage) { return; }
133 
134     var current = this.getDataAtToken(token);
135     goog.array.remove(current, data);
136 
137     // Return the modified data to local storage
138     localStorage.setItem(token, goog.json.serialize(current));
139 };

141 /**
142  * Retrieves a list of show IDs representing the user's subscriptions.
143  * @return {Array.<string>} A list of show IDs representing the user's subscriptions.
144  */
145 gva.model.UserModel.prototype.getUserSubscriptions = function() {
146     return this.getDataAtToken(gva.model.UserModel.TOKEN_SUBSCRIPTIONS);
147 };
148 
149 /**
150  * Subscribes the user to a show.
151  * @param {string} showID The ID of the show to subscribe the user to.
152  */
153 gva.model.UserModel.prototype.addUserSubscription = function(showID) {
154 
155     // Add the subscription to local storage
156     this.addValueToToken(gva.model.UserModel.TOKEN_SUBSCRIPTIONS, showID);
157 
158     // Update the data in the model
159     var dataModel = gva.model.DataModel.getInstance();
160     var show = dataModel.getShowByID(showID);
161     show.subscribed = true;
162     // console.log('added sub to data model');
163 
164     // Notify observers that data has changed
165     var event = new goog.events.Event(gva.model.UserModel.EventType.SUBSCRIPTION_ADDED);
166     event.data = show;
167 
168     this.dispatchEvent(event);
169 };
170 
171 /**
172  * Unsubscribes the user from a given show.
173  * @param {string} showID The ID of the show to unsubscribe from.
174  */
175 gva.model.UserModel.prototype.removeUserSubscription = function(showID) {
176 
177     // Remove the subscription to local storage
178     this.removeValueFromToken(gva.model.UserModel.TOKEN_SUBSCRIPTIONS, showID);
179 
180     // Update the data in the model
181     var dataModel = gva.model.DataModel.getInstance();
182     var show = dataModel.getShowByID(showID);
183     show.subscribed = false;
184 
185     // Notify observers that data has changed
186     var event = new goog.events.Event(gva.model.UserModel.EventType.SUBSCRIPTION_REMOVED);
187     event.data = show;
188 
189     this.dispatchEvent(event);
190 };

193  * Retrieves a list of the episodes a user has already watched
194  * @return {Array.<string>} A list of episode IDs.
195  */
196 gva.model.UserModel.prototype.getWatchedEpisodes = function() {

198 };

200 /**
201  * Marks a given episode as watched
202  * @param {string} episodeID The ID of the episode to mark as watched.
203  */
204 gva.model.UserModel.prototype.setWatchedEpisode = function(episodeID) {
205 
206     // Store this episode as watched in local storage
207     this.addValueToToken(gva.model.UserModel.TOKEN_WATCHED_EPISODES, episodeID);
208 
209     // Update the data in the model
210     var dataModel = gva.model.DataModel.getInstance();
211     var episode = dataModel.getEpisodeByID(episodeID);
212     episode['watched'] = true;
213 
214     // Notify observers that data has changed
215     var event = new goog.events.Event(gva.model.UserModel.EventType.EPISODE_WATCHED);
216     event.data = episode;
217 
218     this.dispatchEvent(event);
219 };

227 };

229 /**
230  * Sets the last episode check time to the current time.
231  */

234     localStorage.setItem(gva.model.UserModel.TOKEN_LAST_EPISODE_CHECK, now);
235 };
