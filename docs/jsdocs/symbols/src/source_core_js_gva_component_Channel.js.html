 39 /**
 40  * @constructor
 41  * @extends {gva.component.BaseComponent}
 42  */
 43 gva.component.Channel = function() {
 44     // Call super constructor
 45     gva.component.BaseComponent.call(this);
 46 
 47     // Episode list container.
 48     this.$episodeList = null;
 49 
 50     // The section element within the list container - used for scrolling logic.
 51     this.$episodeListContainer = null;
 52 
 53     /**
 54      * A list of episode item DOM elements
 55      * @type {Array.<Element>}
 56      */
 57     this.$episodes = [];
 58 
 59     // A list of episode data objects
 60     this.episodeData = {};
 61 
 62     // Component instances to be cached.
 63     this.player = null;
 64     this.userModel = null;
 65 
 66     // Utility for now playing notifications.
 67     this.notifier = new gva.util.Notify();
 68 
 69     // Scrollbar object.
 70     this.scroller = null;
 71 
 72     // Scrollbar element.
 73     this.$scrollerWrap = null;
 74     this.$scrollContainer = null;
 75 
 76     // Handles mousewheel events on episode list.
 77     this.mouseWheelHandler = null;
 78 
 79     // Scrolling interval used while holding up/down scroll buttons.
 80     this.scrollInterval = null;
 81 
 82     /**
 83      * Monitors the viewport size so that the grid can flex.
 84      * @type {goog.dom.ViewportSizeMonitor}
 85      */
 86     this.viewportMonitor = new goog.dom.ViewportSizeMonitor();
 87 
 88     // Sort direction (ascending or descending by airdate).
 89     this.SORT_ORDER = {
 90       ASCENDING: 'ascending',
 91       DESCENDING: 'descending'
 92     };
 93 
 94     this.currentSortOrder = this.SORT_ORDER.DESCENDING;
 95 
 96     this.groupedByShow = true;
 97 
 98     /**
 99      * Enumeration for event types dispatched from this class
100      */
101     gva.component.Channel.EventType = {
102             /**
103              * Dispatched when component needs to get data again (sorting methods)
104              */
105             REFRESH_REQUESTED: 'Channel::refreshRequested'
106     };
107 
108     // Duration of accordion slide animations.
109     this.ACCORDION_SLIDE_DURATION = 350;
110 };

169 /**
170  * Adds an episode to the list based on current sorting/grouping states.
171  * @param {gva.vo.Episode} episode Episode data.
172  */
173 gva.component.Channel.prototype.addEpisode = function(episode) {
174 
175   var episodeNode = this.createEpisodeNode(episode);
176 
177   // Before adding to channel determine the episode's container.
178 
179   // Default to base episode list as container (ungrouped sorting)
180   var $episodeContainer = this.$episodeList;
181 
182   if (this.groupedByShow) {
183 
184     goog.dom.classes.add(this.$episodeList, 'accordion');
185 
186     // Find or create the show's accordion tab.
187     var $showTab = goog.dom.getElement('tab-' + episode.show_id, this.$episodeList);
188 
189     if (!$showTab) {
190 
191       // Create the show's accordion tab if none found.
192       var showTabMarkup = '' +
193         '<li id="tab-' + episode.show_id + '">' +
194           '<a class="show-tab">' +
195             '<h3><span class="icon"></span>' + episode.show_title + '</h3>' +
196           '</a>' +
197           '<ul></ul>' +
198         '</li>';
199 
200       var showTabNode = goog.dom.htmlToDocumentFragment(showTabMarkup);
201       var $accordionButton = goog.dom.getElementByClass('show-tab', showTabNode);
202 
203       // Listen for accordion clicks.
204       goog.events.listen($accordionButton, goog.events.EventType.CLICK, this.onShowTabClicked, false, this);
205 
206       // Add show tab to dom.
207       goog.dom.appendChild(this.$episodeList, showTabNode);
208 
209       $showTab = goog.dom.getElement('tab-' + episode.show_id, this.$episodeList);
210     }
211 
212     // Set destination to episode list in the existing or newly created show tab.
213     $episodeContainer = goog.dom.getElementsByTagNameAndClass('ul', null, $showTab)[0];
214   }
215 
216   // Add to appropriate container.
217   goog.dom.appendChild($episodeContainer, episodeNode);
218 
219   // Store a references
220   this.$episodes.push(episodeNode);
221   this.episodeData[episode.id] = episode;
222 
223   this.refreshScroller();
224 };

227 /**
228  * Creates the episode markup and sets listeners.
229  * @param {gva.vo.Episode} episode Episode data.
230  * @return {Element} The episode element.
231  */
232 gva.component.Channel.prototype.createEpisodeNode = function(episode) {
233   // Format air date.
234   var t = new Date(episode.airdate);
235   var airDate = gva.util.DateUtil.formatDate(t, 'M j, Y');
236 
237   // Trim episode description.
238   var description = gva.util.StringUtil.stripTags(episode.description);
239   description = gva.util.StringUtil.singleSentence(description, 5);
240 
241   var episodeMarkup = '' +
242     '<li id="' + episode.id + '" class="episode-container">' +
243       '<div class="episode">' +
244         '<figure>' +
245           '<img src="' + episode.image_url + '" />' +
246           '<a class="play">play</a>' +
247           '<span>Now Playing</span>' +
248         '</figure>' +
249         '<h3>' + episode.title + '</h3>' +
250         '<p class="description">' + description + '</p>' +
251         '<p class="details">Air Date: ' + airDate + '</p>' +
252         '<a class="show-link" href="#"><span class="icon"></span>' + episode.show_title + '</a>' +
253       '</div>' +
254     '</li>';
255 
256 
257   var episodeNode = null;
258 
259   if (goog.userAgent.IE && goog.userAgent.VERSION < 9) {
260     // Create show element for older browsers.
261     episodeNode = innerShiv(episodeMarkup, false)[0];
262   }
263   else {
264     // Create a DOM element from the markup fragment
265     episodeNode = goog.dom.htmlToDocumentFragment(episodeMarkup);
266   }
267 
268 
269   var $showLink = goog.dom.query('a.show-link', episodeNode)[0];
270 
271   // Configure button attributes
272   var showSlug = gva.util.StringUtil.slugify(episode.show_title);
273   goog.dom.setProperties($showLink, {title: episode.show_title, href: '/shows/' + showSlug + '/'});
274 
275 
276   // Set event listeners.
277 
278   // Show page links.
279   goog.events.listen($showLink, goog.events.EventType.CLICK, this.onShowLinkClicked, false, this);
280 
281   // Play button click listener.
282   var $playBtn = goog.dom.query('.episode figure', episodeNode)[0];
283   goog.events.listen($playBtn, goog.events.EventType.CLICK, this.onPlayClick, false, this);
284 
285   return episodeNode;
286 };

317 /**
318  * Plays an episode and handles all related events including notification, displaying
319  * the episode as 'now playing' in the channel list, and adding the episode to the
320  * channel list if it wasn't found.
321  * @param {gva.vo.Episode} episode Episode data from the feed.
322  */
323 gva.component.Channel.prototype.playEpisode = function(episode) {
324 
325   // Highlight episode in channel.
326   this.setNowPlaying(episode.id);
327 
328   // Start video playback.
329   var self = this;
330   setTimeout(function() {
331     self.player.loadEpisode(episode);
332   }, 500);
333 
334   // Show notification.
335   this.notifier.showNotification('episode', episode);

339 /**
340  * Updates now-playing episode in channel and scrolls it into view.
341  * @param {number} episodeID The episode id to highlight as now playing.
342  */
343 gva.component.Channel.prototype.setNowPlaying = function(episodeID) {
344   // Remove highlight class from previously played episode.
345   var $previousEpisode = goog.dom.getElementByClass('now-playing', this.$episodeList);
346   if ($previousEpisode) {
347     goog.dom.classes.remove($previousEpisode, 'now-playing');
348   }
349 
350   // Find and highlight/scroll to next episode.
351   var $nextEpisode = goog.dom.getElement(episodeID);
352 
353   if (this.groupedByShow) {
354     // Make sure that tab's accordion is open.
355     var $nextEpisodeShowTab = $nextEpisode.parentNode.parentNode;
356 
357     if (!goog.dom.classes.has($nextEpisodeShowTab, 'expanded')) {
358       this.toggleTab($nextEpisodeShowTab, false);
359     }
360   }
361 
362   // Highlight next episode.
363   goog.dom.classes.add($nextEpisode, 'now-playing');
364 
365   // Scroll next episode into view.
366   var scroll = new goog.fx.dom.Scroll(this.$episodeListContainer,
367       [0, this.$episodeListContainer.scrollTop],
368       [0, $nextEpisode.offsetTop],
369       150, goog.fx.easing.easeOut);
370   scroll.play();
371 
372   goog.events.listen(scroll, goog.fx.Transition.EventType.END, goog.bind(this.refreshScroller, this));
373 };

400 /**
401  * Toggles the channel playlist expanded state.
402  */
403 gva.component.Channel.prototype.toggleChannels = function() {
404 
405   // Get the width of the channel component to determine how far we need to slide it.
406   var channelWidth = goog.style.getBorderBoxSize(this.$context).width;
407 
408   // Slide channel and player components left and toggle 'expanded' class.
409   if (goog.dom.classes.has(this.$context, 'expanded')) {
410 
411     // Slide left.
412     goog.style.setStyle(this.$context, 'left', '-' + channelWidth + 'px');
413     goog.style.setStyle(this.player.$context, 'left', '0');
414 
415     // Toggle class.
416     goog.dom.classes.remove(this.$context, 'expanded');
417   }
418   else {
419 
420     // Slide right.
421     goog.style.setStyle(this.$context, 'left', '0');
422     goog.style.setStyle(this.player.$context, 'left', channelWidth + 'px');
423 
424     // Toggle class.
425     goog.dom.classes.add(this.$context, 'expanded');
426   }
427 
428   // Refresh player controls if not playing.
429   goog.Timer.callOnce(this.player.resize, 400, this.player);
430 };

451 /**
452  * Event handler for when a sort by link is clicked.
453  * Updates the sorting options and then calls the sort method.
454  * @param {goog.events.Event} event The event which triggered the handler.
455  */
456 gva.component.Channel.prototype.onSortClick = function(event) {
457 
458   var $clickedBtn = event.currentTarget;
459 
460   // See what criteria we should use to sort the episodes with.
461   var sortby = goog.dom.dataset.get($clickedBtn, 'sortby');
462 
463   // Determine sort state and update styles.
464   if (sortby == 'airdate') {
465 
466     if (goog.dom.classes.has($clickedBtn, 'ascending')) {
467       goog.dom.classes.remove($clickedBtn, 'ascending');
468       this.currentSortOrder = this.SORT_ORDER.DESCENDING;
469     }
470     else {
471       goog.dom.classes.add($clickedBtn, 'ascending');
472       this.currentSortOrder = this.SORT_ORDER.ASCENDING;
473     }
474   }
475   else {
476     if (goog.dom.classes.has($clickedBtn, 'grouped')) {
477       goog.dom.classes.remove($clickedBtn, 'grouped');
478       this.groupedByShow = false;
479     }
480     else {
481       goog.dom.classes.add($clickedBtn, 'grouped');
482       this.groupedByShow = true;
483     }
484   }
485 
486 
487   this.sortEpisodes();
488 };

491 /**
492  * Sorts the episodes based on current sorting options (grouping and order).
493  * Note that these options must be set prior to calling this function.
494  */
495 gva.component.Channel.prototype.sortEpisodes = function() {
496 
497   // Temp copy of the episode data.
498   /**
499    * @type {Array.<gva.vo.Episode>}
500    */
501   var episodes = [];
502   for (var ep in this.episodeData) {
503     episodes.push(this.episodeData[ep]);
504   }
505 
506   // Remember which episode is currently playing.
507   var $nowPlayingEl = goog.dom.getElementByClass('now-playing', this.$episodeList);
508   var nowPlayingID = null;
509 
510   if ($nowPlayingEl) {
511     nowPlayingID = goog.dom.getElementByClass('now-playing', this.$episodeList).id;
512   }
513 
514 
515   // Clear current list of episodes.
516   this.reset();
517 
518   var i, n;
519 
520   // Set sorting method.
521   var sortMethod = (this.currentSortOrder == this.SORT_ORDER.DESCENDING) ? this.sortByAirdateDescending : this.sortByAirdateAscending;
522 
523   if (this.groupedByShow) {
524 
525     // Separate episodes by show.
526     var episodesByShow = {};
527 
528     for (i = 0, n = episodes.length; i < n; i++) {
529       if (episodesByShow[episodes[i].show_title]) {
530         episodesByShow[episodes[i].show_title].push(episodes[i]);
531       }
532       else {
533         episodesByShow[episodes[i].show_title] = [];
534         episodesByShow[episodes[i].show_title].push(episodes[i]);
535       }
536     }
537 
538     // Empty original episode list before merging newly grouped/sorted episodes.
539     episodes = [];
540 
541     // Sort each grouping and merge.
542     for (var show in episodesByShow) {
543       episodesByShow[show].sort(sortMethod);
544       episodes = episodes.concat(episodesByShow[show]);
545     }
546   }
547 
548   else {
549     // Sort without grouping.
550     episodes.sort(sortMethod);
551   }
552 
553   // Repopulate channel playlist.
554   for (i = 0, n = episodes.length; i < n; i++) {
555     this.addEpisode(episodes[i]);
556   }
557 
558   // Reset now playing highlight to original episode.
559   if (nowPlayingID) {
560    this.setNowPlaying(nowPlayingID);
561   }
562 };

565 /**
566  * Sorts function for video data by airdate in descending order.
567  * @param {object} a Video object.
568  * @param {object} b Video object.
569  * @return {Number} The difference (a - b) between video airdate timestamps.
570  */
571 gva.component.Channel.prototype.sortByAirdateAscending = function(a, b) {
572   var aTime = new Date(a.airdate);
573   var bTime = new Date(b.airdate);
574 
575   return (aTime.getTime() - bTime.getTime());
576 };

582  * @param {object} b Video object.

586   var aTime = new Date(a.airdate);
587   var bTime = new Date(b.airdate);
588 
589   return (bTime.getTime() - aTime.getTime());
590 };

593 /**
594  * End video playback handler.
595  */
596 gva.component.Channel.prototype.onEpisodeComplete = function() {
597 
598   // Move to next video in playlist.
599   var $currentEpisode = goog.dom.getElementByClass('now-playing', this.$episodeList);
600   var $nextEpisode = null;
601 
602   if (!$currentEpisode) {
603     // Grab the first video in the list.
604     $nextEpisode = goog.dom.getElementByClass('episode-container', this.$episodeList);
605   }
606   else {
607     
608     // Mark current episode as watched in user model.
609     this.userModel.setWatchedEpisode($currentEpisode.id);
610     
611     // Lookup next video in stored video array.
612     var currentIndex = goog.array.indexOf(this.$episodes, $currentEpisode);
613     
614     var nextIndex = currentIndex + 1;
615     if ((currentIndex == -1) || (currentIndex == this.$episodes.length - 1)) {
616       nextIndex = 0;
617     }
618 
619     $nextEpisode = goog.dom.getElement(this.$episodes[nextIndex]['id']);
620   }
621 
622 
623   var episode = this.episodeData[$nextEpisode['id']];
624   gva.controller.ApplicationController.navigate('/home/' + episode.slug, episode.title);
625 };

637 /**
638  * Toggle accordion open/close with optional animation.
639  * @param {Element} $tab The li tab element to toggle.
640  * @param {Boolean} animated Optional animation control - set to false to expand without slide animation.
641  */
642 gva.component.Channel.prototype.toggleTab = function($tab, animated) {
643   if (!animated) {
644     goog.dom.classes.toggle($tab, 'expanded');
645     this.refreshScroller();
646   }
647 
648   else {
649 
650     var expanding = !goog.dom.classes.has($tab, 'expanded');
651 
652     // Get start/end sizings.
653     var start = goog.style.getBounds($tab);
654     goog.dom.classes.toggle($tab, 'expanded');
655     var end = goog.style.getBounds($tab);
656     goog.dom.classes.toggle($tab, 'expanded');
657 
658 
659     var anim = new goog.fx.dom.Resize($tab,
660       [start.width, start.height],
661       [end.width, end.height],
662       this.ACCORDION_SLIDE_DURATION,
663       gva.easing.quintic.easeOut
664     );
665 
666     // Show content before animating if expanding.
667     goog.events.listen(anim, goog.fx.Transition.EventType.BEGIN, function() {
668       if (expanding) {
669        goog.dom.classes.add($tab, 'expanded');
670       }
671       else {
672         goog.dom.classes.add($tab, 'closing');
673       }
674     });
675 
676     // Clear height styles and toggle hidden if closing.
677     goog.events.listen(anim, goog.fx.Transition.EventType.END, goog.bind(function() {
678       goog.style.setStyle($tab, 'height', '');
679       if (!expanding) {
680        goog.dom.classes.remove($tab, 'expanded');
681        goog.dom.classes.remove($tab, 'closing');
682       }
683 
684       this.refreshScroller();
685 
686     }, this));
687 
688     anim.play();
689   }
690 
691 };

694 /**
695  * Creates and sets event listeners for the scrollbar.
696  */
697 gva.component.Channel.prototype.createScroller = function() {
698 
699   // Create and bind seek slider.
700   this.scroller = new goog.ui.Slider();
701   this.scroller.setOrientation(goog.ui.Slider.Orientation.VERTICAL);
702   this.scroller.orientation_ = 'vertical';
703   this.scroller.decorate(this.$scrollerWrap);
704   this.scroller.setValue(100);
705   this.scroller.setMoveToPointEnabled(true);
706 
707   // Listen for changes to the scroller.
708   goog.events.listen(this.scroller,
709     goog.events.EventType.CHANGE, this.onScrollChange,
710     false, this);
711 
712   // Listen for scroll button up/down interaction.
713   var scrollButtons = goog.dom.getElementsByClass('scroll-button', this.$context);
714 
715   for (var i = 0, n = scrollButtons.length; i < n; i++) {
716 
717     var button = scrollButtons[i];
718 
719     goog.events.listen(button,
720       goog.events.EventType.MOUSEDOWN, this.onScrollButtonMousedown,
721       false, this);
722 
723     goog.events.listen(button,
724       goog.events.EventType.MOUSEUP, this.onScrollButtonStop,
725       false, this);
726   }
727 
728   // Listen for mousewheel events over the content area.
729   var MouseWheelHandler = goog.events.MouseWheelHandler;
730   var MOUSEWHEEL = MouseWheelHandler.EventType.MOUSEWHEEL;
731 
732   this.mouseWheelHandler = new MouseWheelHandler(this.$episodeListContainer);
733 
734   goog.events.listen(this.mouseWheelHandler, MOUSEWHEEL,
735     this.handleMouseWheel, false, this);
736 };

739 /**
740  * Refreshes position of scrollbar to accomodate changed content.
741  */
742 gva.component.Channel.prototype.onScrollChange = function() {
743 
744   var scrollValue = this.scroller.getValue();
745   var contentHeight = goog.style.getSize(this.$episodeList).height;
746   var containerHeight = goog.style.getSize(this.$episodeListContainer).height;
747 
748   // Calculate what proportion of the content height to scroll.
749   var scrollTo = (1 - scrollValue / 100) * (contentHeight - containerHeight);
750   scrollTo = scrollTo.toFixed(2);
751 
752   this.$episodeListContainer.scrollTop = scrollTo;
753 };

756 /**
757  * Refreshes position of scrollbar to accomodate changed content.
758  */
759 gva.component.Channel.prototype.refreshScroller = function() {
760   var scrollOffset = this.$episodeListContainer.scrollTop;
761   var contentHeight = goog.style.getSize(this.$episodeList).height;
762   var containerHeight = goog.style.getSize(this.$episodeListContainer).height;
763   var scrollableHeight = contentHeight - containerHeight;
764 
765   // Calculate the proportional value (0-100) to set the scroller to.
766   var adjustedValue = 100 * (1 - (scrollOffset / scrollableHeight).toFixed(4));
767 
768   this.scroller.setValue(adjustedValue);
769 
770   // Hide scroller thumb if no scrolling can be done.
771   if (contentHeight <= containerHeight) {
772     goog.style.showElement(this.$scrollContainer, false);
773   }
774   else {
775     goog.style.showElement(this.$scrollContainer, true);
776   }
777 };

789 /**
790  * Handles a mousedown on the scroller up/down buttons
791  * @param {goog.events.BrowserEvent} event The event which triggered the handler.
792  */
793 gva.component.Channel.prototype.onScrollButtonMousedown = function(event) {
794 
795   var direction = goog.dom.classes.has(event.currentTarget, 'up') ? 1 : -1;
796   var contentHeight = Math.max(1, goog.style.getSize(this.$episodeList).height);
797   var containerHeight = goog.style.getSize(this.$episodeListContainer).height;
798 
799   // Scroll more for less content since thumb doesn't scale.
800   var step = Math.round(direction * 8 * containerHeight / contentHeight);
801   step = (step == 0) ? direction : step;
802 
803   // Start scrolling.
804   this.scrollInterval = setInterval(goog.bind(function() {
805     var nextValue = this.scroller.getValue() + step;
806 
807     // Check bounds.
808     nextValue = Math.min(100, nextValue);
809     nextValue = Math.max(0, nextValue);
810 
811     this.scroller.setValue(nextValue);
812   }, this), 100);
813 
814   // Listen for mouseout to stop scrolling (mouseup is already being listened for).
815   goog.events.listenOnce(event.currentTarget,
816     goog.events.EventType.MOUSEOUT, this.onScrollButtonStop,
817     false, this);
818 };
