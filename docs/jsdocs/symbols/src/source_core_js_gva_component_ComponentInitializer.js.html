 17 /**
 18  * Provide a namespace for this class
 19  */
 20 goog.provide('gva.component.ComponentInitializer');
 21 
 22 /**
 23  * Import dependencies
 24  */
 25 goog.require('goog.array');
 26 goog.require('goog.dom');
 27 goog.require('goog.dom.dataset');
 28 goog.require('goog.dom.query');
 29 /**
 30  * Closure doesn't allow for pure dependency injection, so we
 31  * need to make reference to the available components
 32  */
 33 goog.require('gva.component.Channel');
 34 goog.require('gva.component.EpisodeGrid');
 35 goog.require('gva.component.Html5Player');
 36 goog.require('gva.component.SearchForm');
 37 goog.require('gva.component.SearchGrid');
 38 goog.require('gva.component.SectionMenu');
 39 goog.require('gva.component.ShowDetails');
 40 goog.require('gva.component.ShowGrid');
 41 goog.require('gva.component.Subscriptions');
 42 goog.require('gva.component.VideoPlayer');
 43 
 44 
 45 /**
 46  * @constructor
 47  */
 48 gva.component.ComponentInitializer = (function() {
 49 
 50     /**
 51      * The key for the data attribute used to identify component contexts.
 52      * @const
 53      * @type {string}
 54      */
 55     var IDENTIFIER_KEY = 'component_id';
 56 
 57     /**
 58      * Simple GUID implementation which increments when used.
 59      * @type {string}
 60      */
 61     var GUID = Math.floor(Math.random() * 0xFFFFFF);
 62 
 63     /**
 64      * A list of all initialized components.
 65      * @type Array.<gva.controller.AbstractComponent>
 66      * @private
 67      */
 68     var _initializedComponents = [];
 69 
 70     /**
 71      * Lookups for initialized components.
 72      */
 73     var _componentIndex = {
 74 
 75             /**
 76              * Components lists indexed by their type
 77              */
 78             byType: {},
 79 
 80             /**
 81              * Component instances indexed by their IDs
 82              */
 83             byID: {}
 84     };
 85 
 86     /**
 87      * Public API
 88      */
 89     return {
 90 
 91         /**
 92          * Traverses the DOM and initializes all found components.
 93          */
 94         registerComponents: function() {
 95 
 96             // Get all DOM elements with a component attribute.
 97             var $components = goog.dom.query('[data-component]');
 98             var type;
 99 
100             // Initialize each DOM element with a component type.
101             goog.array.forEach($components, function(element, index) {
102 
103                 // Retreive the component type.
104                 type = goog.dom.dataset.get(element, 'component');
105 
106                 // Attempt to initialize the component.
107                 gva.component.ComponentInitializer.initializeComponent(type, element);
108             });
109         },
110 
111         /**
112          * Initializes a component of a certain type on a given context.
113          * @param {string} type The class name of the component to initialize.
114          * @param {Element} $context The DOM element which will serve as the component's context.
115          * @return {gva.controller.AbstractComponent} The initialized component instance.
116          */
117         initializeComponent: function(type, $context) {
118 
119             // If the component has an ID, it's already been initialized.
120             var hasID = goog.dom.dataset.has($context, IDENTIFIER_KEY);
121             var component;
122 
123             if (hasID) {
124 
125                 // Component is already initialized.
126                 var givenID = goog.dom.dataset.get($context, IDENTIFIER_KEY);
127 
128                 // Just return the initialized component.
129                 component = _componentIndex.byID[givenID];
130 
131             } else {
132 
133                 // Retrieve the function definition for the component.
134                 var ComponentDefinition = goog.getObjectByName(type);
135 
136                 if (typeof ComponentDefinition === 'function') {
137 
138                     // Create an instance of the component class.
139                     component = new ComponentDefinition();
140 
141                     // Get next GUID.
142                     GUID++;
143 
144                     // Tag the DOM element with it's component ID.
145                     goog.dom.dataset.set($context, IDENTIFIER_KEY, GUID);
146 
147                     // Index the component so we can retrieve it easily.
148                     _componentIndex.byID[GUID] = component;
149 
150                     // Keep track of all instances of this type of component.
151                     _componentIndex.byType[type] = _componentIndex.byType[type] || [];
152                     _componentIndex.byType[type].push(component);
153 
154                     // Initialize the component.
155                     component.initialize($context);
156 
157                     // Creates some indexed references to the component.
158                     _initializedComponents.push(component);
159 
160                 } else {
161                     console.warn('Component definition "' + type + '" not found or is not a Function');
162                 }
163             }
164 
165             return component;
166         },
167 
168         /**
169          * Retrieves a component instance by ID.
170          * @param {string} id The ID of the component to retrieve.
171          * @return {gva.controller.AbstractComponent} The component instance.
172          */
173         getComponentByID: function(id) {
174             return _componentIndex.byID[id];
175         },
176 
177         /**
178          * Retrieves a component instance based on it's DOM element (context).
179          * @param {Element} $element The DOM element (context) of the component.
180          * @return {gva.controller.AbstractComponent} The component instance.
181          */
182         getComponentByElement: function($element) {
183             var id = goog.dom.dataset.get($element, IDENTIFIER_KEY);
184             return gva.component.ComponentInitializer.getComponentByID(id);
185         },
186 
187         /**
188          * Retrieves a component instance based on a DOM query.
189          * @param {string} query A DOM query, for example "#container > .item".
190          * @return {gva.controller.AbstractComponent} The component instance.
191          */
192         getComponentByQuery: function(query) {
193             var element = goog.dom.query(query)[0];
194             return gva.component.ComponentInitializer.getComponentByElement(element);
195         }
196 
197     };
198 
199 })();
