 36 /**
 37  * Creates a new FlexibleGrid instance.
 38  * @constructor
 39  */
 40 gva.component.FlexibleGrid = function() {
 41 
 42     this.$context = null;
 43 
 44     /**
 45      * The grid boundaries.
 46      * @type {goog.math.Rect}
 47      */
 48     this._bounds = null;
 49 
 50     this._currentIndex = 0;
 51     this._currentPage = 0;
 52     this._totalPages = 1;
 53 
 54     /**
 55      * A list of grid items DOM elements
 56      * @type {Array.<Element>}
 57      */
 58     this._$items = [];
 59 
 60     this._$prevBtn = null;
 61     this._$nextBtn = null;
 62 
 63     /**
 64      * Monitors the viewport size so that the grid can flex.
 65      * @type {goog.dom.ViewportSizeMonitor}
 66      */
 67     this._viewportMonitor = new goog.dom.ViewportSizeMonitor();
 68 
 69     /**
 70      * Enumeration for event types dispatched from this class
 71      */
 72     gva.component.FlexibleGrid.EventType = {
 73             /**
 74              * Dispatched whenever grid is refreshed.
 75              */
 76             REFRESHED: 'FlexibleGrid::refreshed'
 77     };
 78 
 79     // Call super constructor.
 80     gva.util.Grid.call(this);
 81 };
 82 
 83 /**
 84  * Extend gva.util.Grid
 85  */
 86 goog.inherits(gva.component.FlexibleGrid, gva.util.Grid);

 89 /**
 90  * Initializes a new grid.
 91  * @param {Element} $context The context element the grid is placed in.
 92  * @param {number} cellWidth The width of each cell.
 93  * @param {number} cellHeight The height of each cell.
 94  * @param {number} cellSpacingX The horizontal space between each column.
 95  * @param {number} cellSpacingY The vertical space between each row.
 96  */
 97 gva.component.FlexibleGrid.prototype.initialize = function($context, cellWidth, cellHeight, cellSpacingX, cellSpacingY) {
 98 
 99     this.$context = $context;
100 
101     // Build the pagination menu
102     this.createNavigation();
103 
104     // Bind event handlers
105     this.addListeners();
106 
107     // Set up the fixed grid cell size.
108     this.constructor.superClass_.setCellBounds.call(this, cellWidth, cellHeight, cellSpacingX, cellSpacingY);
109 
110     // Resize to initial boundaries.
111     this.resize();
112 };
113 
114 /**
115  * Creates the pagination navigation and appends it to the context
116  */
117 gva.component.FlexibleGrid.prototype.createNavigation = function() {
118 
119     // Create a navigation node
120     var $nav = goog.dom.createDom('nav');
121     goog.dom.classes.add($nav, 'pagination');
122 
123     // Create next and previous buttons
124     this._$prevBtn = goog.dom.createDom('a', {'href': '#'}, 'Previous Page');
125     this._$nextBtn = goog.dom.createDom('a', {'href': '#'}, 'Next Page');
126 
127     goog.dom.classes.add(this._$prevBtn, 'prev');
128     goog.dom.classes.add(this._$nextBtn, 'next');
129 
130     // Add icons to them
131     goog.dom.appendChild(this._$prevBtn, goog.dom.createDom('span', {'class': 'icon'}));
132     goog.dom.appendChild(this._$nextBtn, goog.dom.createDom('span', {'class': 'icon'}));
133 
134     // Add buttons to nav
135     goog.dom.appendChild($nav, this._$prevBtn);
136     goog.dom.appendChild($nav, this._$nextBtn);
137 
138     // Add all to the DOM
139     goog.dom.appendChild(this.$context, $nav);
140 };
141 
142 /**
143  * Binds event listeners to viewport resize and pagination mouse events
144  */
145 gva.component.FlexibleGrid.prototype.addListeners = function() {
146 
147     // Listen for window resize in order to update the grid.
148     goog.events.listen(this._viewportMonitor, goog.events.EventType.RESIZE, goog.bind(this.resize, this));
149 
150     // Listen for clicks on pagination links
151     goog.events.listen(this._$prevBtn, goog.events.EventType.CLICK, goog.bind(this.onNavClick, this));
152     goog.events.listen(this._$nextBtn, goog.events.EventType.CLICK, goog.bind(this.onNavClick, this));
153 };

155 /**
156  * Adds a DOM element to the grid
157  * @param {Element} $item The DOM element to add to the grid.
158  */
159 gva.component.FlexibleGrid.prototype.addItem = function($item) {
160     // Append the item to the grid
161     goog.dom.appendChild(this.$context, $item);
162 
163     // Store a reference
164     this._$items.push($item);
165 
166     // Refresh the grid to accommodate the new item
167     this.refresh(false);
168 };
169 
170 /**
171  * Returns a list of all items controlled by the grid, including those that
172  * might not be visible in the current page
173  * @return {Array.<Element>} All items (visible and hidden) in this grid.
174  */
175 gva.component.FlexibleGrid.prototype.getItems = function() {
176     return this._$items;
177 };
178 
179 /**
180  * Removes all items from the grid and reset the current page and offset
181  */
182 gva.component.FlexibleGrid.prototype.clear = function() {
183 
184     // Remove any current items
185     goog.array.forEach(this._$items, function($item) {
186         goog.dom.removeNode($item);
187     });
188 
189     // Go back to the first page
190     this._currentIndex = 0;
191     this._currentPage = 0;
192 
193     // Clear the items array
194     this._$items = [];
195 };
196 
197 /**
198  * Navigates to the previous grid page
199  */
200 gva.component.FlexibleGrid.prototype.prevPage = function() {
201     if (this._currentIndex > 0) {
202         this._currentIndex -= this.cells;
203         if (this._currentIndex < 0) {
204           this._currentIndex = 0;
205         }
206         this.refresh(true, 'ltr');
207     }
208 };
209 
210 /**
211  * Navigates to the next grid page
212  */
213 gva.component.FlexibleGrid.prototype.nextPage = function() {
214     if (this._currentIndex < this._$items.length - this.cells) {
215         this._currentIndex += this.cells;
216         this.refresh(true, 'rtl');
217     }
218 };
219 
220 /**
221  * Refreshes the grid view and updates visible items, with an optional transition effect
222  * @param {boolean} transition Whether to use a transition when switching items.
223  * @param {string} direction The horizontal transition direction 'ltr' or 'rtl'.
224  */
225 gva.component.FlexibleGrid.prototype.refresh = function(transition, direction) {
226 
227     transition = !!transition;
228 
229     // Refresh the super grid
230     this.constructor.superClass_.refresh.call(this);
231 
232     // Compute the number of pages
233     this._totalPages = Math.ceil(this._$items.length / this.cells);
234 
235     // Calculate which items will be on screen
236     var min = this._currentIndex;
237     var max = min + this.cells;
238 
239     var windowWidth = this._viewportMonitor.getSize().width;
240 
241     // Reposition elements.
242     var $item, coord;
243 
244     for (var i = 0, j = 0, n = this._$items.length; i < n; i++) {
245 
246         $item = this._$items[i];
247 
248         if (i >= min && i < max) {
249 
250             coord = this.coords[j++];
251 
252             // Show and position the element if it should appear in the grid
253 
254             if (transition) {
255 
256               // Position one screen left or right of final position and then slide to final position.
257               var start = (direction == 'ltr') ? [coord.x - windowWidth, coord.y] : [coord.x + windowWidth, coord.y];
258               var end = [coord.x, coord.y];
259 
260               var slide = new goog.fx.dom.Slide($item, start, end, 250, goog.fx.easing.easeOut);
261 
262               // Show element at the beginning of slide.
263               goog.events.listen(slide, goog.fx.Animation.EventType.BEGIN,
264                 function(event) {
265                   goog.style.showElement(event.currentTarget.element, true);
266               });
267 
268               slide.play();
269             }
270 
271             else {
272               goog.style.showElement($item, true);
273               goog.style.setPosition($item, coord.x, coord.y);
274             }
275 
276         } else {
277 
278             // Hide the element if it doesn't appear in the grid
279 
280             if (transition && goog.style.isElementShown($item)) {
281 
282               // Slide from current position one screen left or right.
283               coord = goog.style.getPosition($item);
284               var start = [coord.x, coord.y];
285               var end = (direction == 'ltr') ? [coord.x + windowWidth, coord.y] : [coord.x - windowWidth, coord.y];
286 
287               var slide = new goog.fx.dom.Slide($item, start, end, 250, goog.fx.easing.easeOut);
288 
289               // Hide element once off screen at the end of slide.
290               goog.events.listen(slide, goog.fx.Animation.EventType.END,
291                 function(event) {
292                   goog.style.showElement(event.currentTarget.element, false);
293               });
294 
295               slide.play();
296             }
297 
298             else {
299               goog.style.showElement($item, false);
300             }
301         }
302     }
303 
304     // Disable previous button if necessary
305     if (min < 1) {
306         goog.dom.classes.add(this._$prevBtn, 'disabled');
307     } else {
308         goog.dom.classes.remove(this._$prevBtn, 'disabled');
309     }
310 
311     // Disable next button if necessary
312     if (max > this._$items.length) {
313         goog.dom.classes.add(this._$nextBtn, 'disabled');
314     } else {
315         goog.dom.classes.remove(this._$nextBtn, 'disabled');
316     }
317 
318 
319     // Dispatch refresh event for listeners.
320     var event = new goog.events.Event(gva.component.FlexibleGrid.EventType.REFRESHED);
321     // event.data = this._$items[this._currentIndex];
322     this.dispatchEvent(event);
323 };

325 /**
326  * Updates grid boundaries and refreshes the grid if necessary.
327  * @param {goog.events.Event=} event The event that triggered the resize.
328  */
329 gva.component.FlexibleGrid.prototype.resize = function(event) {
330 
331     // Get the current bounds of the DOM context.
332     this._bounds = goog.style.getBounds(this.$context);
333 
334     // Update grid boundaries to force recalculation.
335     this.constructor.superClass_.setGridBounds.call(this, 0, 0, this._bounds.width, this._bounds.height);
336 
337     // Refresh the grid
338     this.refresh(false);
339 };

341 /**
342  * Event handler for when an pagination link is clicked
343  * @param {goog.events.Event} event The event which triggered the handler.
344  */
345 gva.component.FlexibleGrid.prototype.onNavClick = function(event) {
346     if (event.currentTarget === this._$prevBtn) {
347         this.prevPage();
348     } else {
349         this.nextPage();
350     }
351     event.preventDefault();
352 };
